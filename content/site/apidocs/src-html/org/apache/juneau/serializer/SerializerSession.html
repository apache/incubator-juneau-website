027/**
028 * Context object that lives for the duration of a single use of {@link Serializer}.
029 * <p>
030 * Used by serializers for the following purposes:
031 * <ul class='spaced-list'>
032 *    <li>Keeping track of how deep it is in a model for indentation purposes.
033 *    <li>Ensuring infinite loops don't occur by setting a limit on how deep to traverse a model.
034 *    <li>Ensuring infinite loops don't occur from loops in the model (when detectRecursions is enabled.
035 *    <li>Allowing serializer properties to be overridden on method calls.
036 * </ul>
037 * <p>
038 * This class is NOT thread safe.  It is meant to be discarded after one-time use.
039 */
040public class SerializerSession extends BeanSession {

057   private final UriContext uriContext;

073   /**
074    * Create a new session using properties specified in the context.
075    *
076    * @param ctx The context creating this session object.
077    *    The context contains all the configuration settings for this object.
078    * @param output The output object.
079    *    <br>Character-based serializers can handle the following output class types:
080    *    <ul>
081    *       <li>{@link Writer}
082    *       <li>{@link OutputStream} - Output will be written as UTF-8 encoded stream.
083    *       <li>{@link File} - Output will be written as system-default encoded stream.
084    *    </ul>
085    *    <br>Stream-based serializers can handle the following output class types:
086    *    <ul>
087    *       <li>{@link OutputStream}
088    *       <li>{@link File}
089    *    </ul>
090    * @param op The override properties.
091    *    These override any context properties defined in the context.
092    * @param javaMethod The java method that called this serializer, usually the method in a REST servlet.
093    * @param locale The session locale.
094    *    If <jk>null</jk>, then the locale defined on the context is used.
095    * @param timeZone The session timezone.
096    *    If <jk>null</jk>, then the timezone defined on the context is used.
097    * @param mediaType The session media type (e.g. <js>"application/json"</js>).
098    * @param uriContext The URI context.
099    *    Identifies the current request URI used for resolution of URIs to absolute or root-relative form.
100    */
101   public SerializerSession(SerializerContext ctx, ObjectMap op, Object output, Method javaMethod, Locale locale, TimeZone timeZone, MediaType mediaType, UriContext uriContext) {
102      super(ctx, op, locale, timeZone, mediaType);
103      this.javaMethod = javaMethod;
104      this.output = output;
105      UriResolution uriResolution = null;
106      UriRelativity uriRelativity = null;
107      if (op == null || op.isEmpty()) {
108         maxDepth = ctx.maxDepth;
109         initialDepth = ctx.initialDepth;
110         detectRecursions = ctx.detectRecursions;
111         ignoreRecursions = ctx.ignoreRecursions;
112         useWhitespace = ctx.useWhitespace;
113         addBeanTypeProperties = ctx.addBeanTypeProperties;
114         trimNulls = ctx.trimNulls;
115         trimEmptyCollections = ctx.trimEmptyCollections;
116         trimEmptyMaps = ctx.trimEmptyMaps;
117         trimStrings = ctx.trimStrings;
118         quoteChar = ctx.quoteChar;
119         relativeUriBase = ctx.relativeUriBase;
120         absolutePathUriBase = ctx.absolutePathUriBase;
121         sortCollections = ctx.sortCollections;
122         sortMaps = ctx.sortMaps;
123         abridged = ctx.abridged;
124         uriResolution =  ctx.uriResolution;
125         uriRelativity = ctx.uriRelativity;
126      } else {
127         maxDepth = op.getInt(SERIALIZER_maxDepth, ctx.maxDepth);
128         initialDepth = op.getInt(SERIALIZER_initialDepth, ctx.initialDepth);
129         detectRecursions = op.getBoolean(SERIALIZER_detectRecursions, ctx.detectRecursions);
130         ignoreRecursions = op.getBoolean(SERIALIZER_ignoreRecursions, ctx.ignoreRecursions);
131         useWhitespace = op.getBoolean(SERIALIZER_useWhitespace, ctx.useWhitespace);
132         addBeanTypeProperties = op.getBoolean(SERIALIZER_addBeanTypeProperties, ctx.addBeanTypeProperties);
133         trimNulls = op.getBoolean(SERIALIZER_trimNullProperties, ctx.trimNulls);
134         trimEmptyCollections = op.getBoolean(SERIALIZER_trimEmptyCollections, ctx.trimEmptyCollections);
135         trimEmptyMaps = op.getBoolean(SERIALIZER_trimEmptyMaps, ctx.trimEmptyMaps);
136         trimStrings = op.getBoolean(SERIALIZER_trimStrings, ctx.trimStrings);
137         quoteChar = op.getString(SERIALIZER_quoteChar, ""+ctx.quoteChar).charAt(0);
138         relativeUriBase = op.getString(SERIALIZER_relativeUriBase, ctx.relativeUriBase);
139         absolutePathUriBase = op.getString(SERIALIZER_absolutePathUriBase, ctx.absolutePathUriBase);
140         sortCollections = op.getBoolean(SERIALIZER_sortCollections, ctx.sortMaps);
141         sortMaps = op.getBoolean(SERIALIZER_sortMaps, ctx.sortMaps);
142         abridged = op.getBoolean(SERIALIZER_abridged, ctx.abridged);
143         uriResolution = op.get(UriResolution.class, SERIALIZER_uriResolution, ctx.uriResolution);
144         uriRelativity = op.get(UriRelativity.class, SERIALIZER_uriRelativity, ctx.uriRelativity);
145      }
146      this.uriContext = uriContext != null ? uriContext : new UriContext(uriResolution, uriRelativity, null, null, null, null);
147
148      this.indent = initialDepth;
149      if (detectRecursions || isDebug()) {
150         set = new IdentityHashMap<Object,Object>();
151      } else {
152         set = Collections.emptyMap();
153      }
154   }

156   /**
157    * Wraps the specified output object inside an output stream.
158    * Subclasses can override this method to implement their own specialized output streams.
159    * <p>
160    * This method can be used if the output object is any of the following class types:
161    * <ul>
162    *    <li>{@link OutputStream}
163    *    <li>{@link File}
164    * </ul>
165    *
166    * @return The output object wrapped in an output stream.
167    * @throws Exception If object could not be converted to an output stream.
168    */
169   public OutputStream getOutputStream() throws Exception {
170      if (output == null)
171         throw new SerializeException("Output cannot be null.");
172      if (output instanceof OutputStream)
173         return (OutputStream)output;
174      if (output instanceof File) {
175         if (outputStream == null)
176            outputStream = new BufferedOutputStream(new FileOutputStream((File)output));
177         return outputStream;
178      }
179      throw new SerializeException("Cannot convert object of type {0} to an OutputStream.", output.getClass().getName());
180   }

183   /**
184    * Wraps the specified output object inside a writer.
185    * Subclasses can override this method to implement their own specialized writers.
186    * <p>
187    * This method can be used if the output object is any of the following class types:
188    * <ul>
189    *    <li>{@link Writer}
190    *    <li>{@link OutputStream} - Output will be written as UTF-8 encoded stream.
191    *    <li>{@link File} - Output will be written as system-default encoded stream.
192    * </ul>
193    *
194    * @return The output object wrapped in a Writer.
195    * @throws Exception If object could not be converted to a writer.
196    */
197   public Writer getWriter() throws Exception {
198      if (output == null)
199         throw new SerializeException("Output cannot be null.");
200      if (output instanceof Writer)
201         return (Writer)output;
202      if (output instanceof OutputStream) {
203         if (flushOnlyWriter == null)
204            flushOnlyWriter = new OutputStreamWriter((OutputStream)output, IOUtils.UTF8);
205         return flushOnlyWriter;
206      }
207      if (output instanceof File) {
208         if (writer == null)
209            writer = new OutputStreamWriter(new BufferedOutputStream(new FileOutputStream((File)output)));
210         return writer;
211      }
212      throw new SerializeException("Cannot convert object of type {0} to a Writer.", output.getClass().getName());
213   }

396   /**
397    * Push the specified object onto the stack.
398    *
399    * @param attrName The attribute name.
400    * @param o The current object being serialized.
401    * @param eType The expected class type.
402    * @return The {@link ClassMeta} of the object so that <code>instanceof</code> operations
403    *    only need to be performed once (since they can be expensive).<br>
404    * @throws SerializeException If recursion occurred.
405    */
406   public ClassMeta<?> push(String attrName, Object o, ClassMeta<?> eType) throws SerializeException {
407      indent++;
408      isBottom = true;
409      if (o == null)
410         return null;
411      Class<?> c = o.getClass();
412      ClassMeta<?> cm = (eType != null && c == eType.getInnerClass()) ? eType : getClassMeta(c);
413      if (cm.isCharSequence() || cm.isNumber() || cm.isBoolean())
414         return cm;
415      if (detectRecursions || isDebug()) {
416         if (stack.size() > maxDepth)
417            return null;
418         if (willRecurse(attrName, o, cm))
419            return null;
420         isBottom = false;
421         stack.add(new StackElement(stack.size(), attrName, o, cm));
422         if (isDebug())
423            getLogger().info(getStack(false));
424         set.put(o, o);
425      }
426      return cm;
427   }

429   /**
430    * Returns <jk>true</jk> if {@link SerializerContext#SERIALIZER_detectRecursions} is enabled, and the specified
431    *    object is already higher up in the serialization chain.
432    *
433    * @param attrName The bean property attribute name, or some other identifier.
434    * @param o The object to check for recursion.
435    * @param cm The metadata on the object class.
436    * @return <jk>true</jk> if recursion detected.
437    * @throws SerializeException If recursion occurred.
438    */
439   public boolean willRecurse(String attrName, Object o, ClassMeta<?> cm) throws SerializeException {
440      if (! (detectRecursions || isDebug()))
441         return false;
442      if (! set.containsKey(o))
443         return false;
444      if (ignoreRecursions && ! isDebug())
445         return true;
446
447      stack.add(new StackElement(stack.size(), attrName, o, cm));
448      throw new SerializeException("Recursion occurred, stack={0}", getStack(true));
449   }

451   /**
452    * Pop an object off the stack.
453    */
454   public void pop() {
455      indent--;
456      if ((detectRecursions || isDebug()) && ! isBottom)  {
457         Object o = stack.removeLast().o;
458         Object o2 = set.remove(o);
459         if (o2 == null)
460            addWarning("Couldn't remove object of type ''{0}'' on attribute ''{1}'' from object stack.", o.getClass().getName(), stack);
461      }
462      isBottom = false;
463   }

500   /**
501    * Generalize the specified object if a POJO swap is associated with it.
502    *
503    * @param o The object to generalize.
504    * @param type The type of object.
505    * @return The generalized object, or <jk>null</jk> if the object is <jk>null</jk>.
506    * @throws SerializeException If a problem occurred trying to convert the output.
507    */
508   @SuppressWarnings({ "rawtypes", "unchecked" })
509   public final Object generalize(Object o, ClassMeta<?> type) throws SerializeException {
510      if (o == null)
511         return null;
512      PojoSwap f = (type == null || type.isObject() ? getClassMeta(o.getClass()).getPojoSwap() : type.getPojoSwap());
513      if (f == null)
514         return o;
515      return f.swap(this, o);

518   /**
519    * Returns <jk>true</jk> if the specified value should not be serialized.
520    *
521    * @param cm The class type of the object being serialized.
522    * @param attrName The bean attribute name, or <jk>null</jk> if this isn't a bean attribute.
523    * @param value The object being serialized.
524    * @return <jk>true</jk> if the specified value should not be serialized.
525    * @throws SerializeException If recursion occurred.
526    */
527   public final boolean canIgnoreValue(ClassMeta<?> cm, String attrName, Object value) throws SerializeException {
528
529      if (trimNulls && value == null)
530         return true;
531
532      if (value == null)
533         return false;
534
535      if (cm == null)
536         cm = object();
537
538      if (trimEmptyCollections) {
539         if (cm.isArray() || (cm.isObject() && value.getClass().isArray())) {
540            if (((Object[])value).length == 0)
541               return true;
542         }
543         if (cm.isCollection() || (cm.isObject() && isParentClass(Collection.class, value.getClass()))) {
544            if (((Collection<?>)value).isEmpty())
545               return true;
546         }
547      }
548
549      if (trimEmptyMaps) {
550         if (cm.isMap() || (cm.isObject() && isParentClass(Map.class, value.getClass()))) {
551            if (((Map<?,?>)value).isEmpty())
552               return true;
553         }
554      }
555
556      if (trimNulls && willRecurse(attrName, value, cm))
557         return true;
558
559      return false;
560   }

586   /**
587    * Converts a String to an absolute URI based on the {@link SerializerContext#SERIALIZER_absolutePathUriBase} and
588    *    {@link SerializerContext#SERIALIZER_relativeUriBase} settings on this context.
589    *
590    * @param uri The input URI.
591    * @return The resolved URI.
592    */
593   public String resolveUri(String uri) {
594      if (uri.indexOf("://") != -1 || (absolutePathUriBase == null && relativeUriBase == null))
595         return uri;
596      StringBuilder sb = getStringBuilder();
597      if (StringUtils.startsWith(uri, '/')) {
598         if (absolutePathUriBase != null)
599            sb.append(absolutePathUriBase);
600      } else {
601         if (relativeUriBase != null) {
602            sb.append(relativeUriBase);
603            if (! uri.equals("/"))
604               sb.append("/");
605         }
606      }
607      sb.append(uri);
608      String s = sb.toString();
609      returnStringBuilder(sb);
610      return s;
611   }

630   @Override
631   public boolean close() {
632      if (super.close()) {
633         try {
634            if (outputStream != null)
635               outputStream.close();
636            if (flushOnlyWriter != null)
637               flushOnlyWriter.flush();
638            if (writer != null)
639               writer.close();
640         } catch (IOException e) {
641            throw new BeanRuntimeException(e);
642         }
643         return true;
644      }
645      return false;
646   }

652      private ClassMeta<?> aType;
653
654      private StackElement(int depth, String name, Object o, ClassMeta<?> aType) {
655         this.depth = depth;
656         this.name = name;
657         this.o = o;
658         this.aType = aType;
659      }
660
661      private String toString(boolean simple) {
662         StringBuilder sb = new StringBuilder().append('[').append(depth).append(']');
663         sb.append(StringUtils.isEmpty(name) ? "<noname>" : name).append(':');
664         sb.append(aType.toString(simple));
665         if (aType != aType.getSerializedClassMeta())
666            sb.append('/').append(aType.getSerializedClassMeta().toString(simple));
667         return sb.toString();
668      }
669   }
670
671   private String getStack(boolean full) {
672      StringBuilder sb = new StringBuilder();
673      for (StackElement e : stack) {
674         if (full) {
675            sb.append("\n\t");
676            for (int i = 1; i < e.depth; i++)
677               sb.append("  ");
678            if (e.depth > 0)
679               sb.append("->");
680            sb.append(e.toString(false));
681         } else {
682            sb.append(" > ").append(e.toString(true));
683         }
684      }
685      return sb.toString();
686   }

689    * Returns information used to determine at what location in the parse a failure occurred.
690    *
691    * @return A map, typically containing something like <code>{line:123,column:456,currentProperty:"foobar"}</code>
692    */
693   public Map<String,Object> getLastLocation() {
694      Map<String,Object> m = new LinkedHashMap<String,Object>();
695      if (currentClass != null)
696         m.put("currentClass", currentClass);
697      if (currentProperty != null)
698         m.put("currentProperty", currentProperty);
699      if (stack != null && ! stack.isEmpty())
700         m.put("stack", stack);
701      return m;
702   }

716   /**
717    * Resolves the dictionary name for the actual type.
718    *
719    * @param eType The expected type of the bean property.
720    * @param aType The actual type of the bean property.
721    * @param pMeta The current bean property being serialized.
722    * @return The bean dictionary name, or <jk>null</jk> if a name could not be found.
723    */
724   public String getBeanTypeName(ClassMeta<?> eType, ClassMeta<?> aType, BeanPropertyMeta pMeta) {
725      if (eType == aType)
726         return null;
727
728      if (! isAddBeanTypeProperties())
729         return null;
730
731      String eTypeTn = eType.getDictionaryName();
732
733      // First see if it's defined on the actual type.
734      String tn = aType.getDictionaryName();
735      if (tn != null && ! tn.equals(eTypeTn)) {
736         return tn;
737      }
738
739      // Then see if it's defined on the expected type.
740      // The expected type might be an interface with mappings for implementation classes.
741      BeanRegistry br = eType.getBeanRegistry();
742      if (br != null) {
743         tn = br.getTypeName(aType);
744         if (tn != null && ! tn.equals(eTypeTn))
745            return tn;
746      }
747
748      // Then look on the bean property.
749      br = pMeta == null ? null : pMeta.getBeanRegistry();
750      if (br != null) {
751         tn = br.getTypeName(aType);
752         if (tn != null && ! tn.equals(eTypeTn))
753            return tn;
754      }
755
756      // Finally look in the session.
757      br = getBeanRegistry();
758      if (br != null) {
759         tn = br.getTypeName(aType);
760         if (tn != null && ! tn.equals(eTypeTn))
761            return tn;
762      }
763
764      return null;
765   }
