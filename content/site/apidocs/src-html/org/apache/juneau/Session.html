013package org.apache.juneau;
014
015import java.text.*;
016import java.util.*;
017
018import org.apache.juneau.annotation.*;
019import org.apache.juneau.internal.*;
020import org.apache.juneau.json.*;
021import org.apache.juneau.serializer.*;
022
023/**
024 * A one-time-use non-thread-safe object that's meant to be used once and then thrown away.
025 * <p>
026 * Serializers and parsers use session objects to retrieve config properties and to use it
027 *    as a scratchpad during serialize and parse actions.
028 *
029 * @see PropertyStore
030 */
031public abstract class Session {
032
033   private JuneauLogger logger;
034
035   private final ObjectMap properties;
036   private final Context ctx;
037   private Map<String,Object> cache;
038   private boolean closed;
039   private List<String> warnings;                 // Any warnings encountered.
040
041   /**
042    * Default constructor.
043    *
044    * @param ctx The context creating this session object.
045    * The context contains all the configuration settings for the session.
046    * @param op Properties associated with this session.
047    */
048   protected Session(final Context ctx, ObjectMap op) {
049      this.ctx = ctx;
050      this.properties = op != null ? op : ObjectMap.EMPTY_MAP;
051   }
052
053   /**
054    * Returns the session property with the specified key.
055    * <p>
056    * The order of lookup for the property is as follows:
057    * <ul>
058    *    <li>Override property passed in through the constructor.
059    *    <li>Property defined on the context object.
060    *    <li>System.property.
061    * </ul>
062    * 
063    * @param key The property key.
064    * @return The property value, or <jk>null</jk> if it doesn't exist.
065    */
066   public String getProperty(String key) {
067      return getProperty(key, null);
068   }
069
070   /**
071    * Same as {@link #getProperty(String)} but with a default value.
072    * 
073    * @param key The property key.
074    * @param def The default value if the property doesn't exist or is <jk>null</jk>.
075    * @return The property value.
076    */
077   public String getProperty(String key, String def) {
078      Object v = properties.get(key);
079      if (v == null)
080         v = ctx.getPropertyStore().getProperty(key, String.class, null);
081      if (v == null)
082         v = def;
083      return StringUtils.toString(v);
084   }
085
086   /**
087    * Same as {@link #getProperty(String)} but transforms the value to the specified type.
088    * 
089    * @param type The class type of the value.
090    * @param key The property key.
091    * @return The property value.
092    */
093   public <T> T getProperty(Class<T> type, String key) {
094      return getProperty(type, key, null);
095   }
096
097   /**
098    * Same as {@link #getProperty(Class,String)} but with a default value.
099    * 
100    * @param type The class type of the value.
101    * @param key The property key.
102    * @param def The default value if the property doesn't exist or is <jk>null</jk>.
103    * @return The property value.
104    */
105   public <T> T getProperty(Class<T> type, String key, T def) {
106      T t = properties.get(type, key);
107      if (t == null)
108         t = ctx.getPropertyStore().getProperty(key, type, def);
109      return t;
110   }
111
112   /**
113    * Adds an arbitrary object to this session's cache.
114    * Can be used to store objects for reuse during a session.
115    *
116    * @param key The key.  Can be any string.
117    * @param val The cahed object.
118    */
119   public void addToCache(String key, Object val) {
120      if (cache == null)
121         cache = new TreeMap<String,Object>();
122      cache.put(key, val);
123   }
124
125   /**
126    * Adds arbitrary objects to this session's cache.
127    * Can be used to store objects for reuse during a session.
128    *
129    * @param cacheObjects The objects to add to this session's cache.
130    * No-op if <jk>null</jk>.
131    */
132   public void addToCache(Map<String,Object> cacheObjects) {
133      if (cacheObjects != null) {
134         if (cache == null)
135            cache = new TreeMap<String,Object>();
136         cache.putAll(cacheObjects);
137      }
138   }
139
140   /**
141    * Returns an object stored in the session cache.
142    *
143    * @param c The class type of the object.
144    * @param key The session object key.
145    * @return The cached object, or <jk>null</jk> if it doesn't exist.
146    */
147   @SuppressWarnings("unchecked")
148   public <T> T getFromCache(Class<T> c, String key) {
149      return cache == null ? null : (T)cache.get(key);
150   }
151
152   /**
153    * Logs a warning message.
154    *
155    * @param msg The warning message.
156    * @param args Optional {@link MessageFormat}-style arguments.
157    */
158   public final void addWarning(String msg, Object... args) {
159      if (warnings == null)
160         warnings = new LinkedList<String>();
161      getLogger().warning(msg, args);
162      msg = args.length == 0 ? msg : MessageFormat.format(msg, args);
163      warnings.add((warnings.size() + 1) + ": " + msg);
164   }
165
166   /**
167    * Returns <jk>true</jk> if warnings occurred in this session.
168    *
169    * @return <jk>true</jk> if warnings occurred in this session.
170    */
171   public boolean hasWarnings() {
172      return warnings != null && warnings.size() > 0;
173   }
174
175   /**
176    * Returns the warnings that occurred in this session.
177    *
178    * @return The warnings that occurred in this session, or <jk>null</jk> if no warnings occurred.
179    */
180   public List<String> getWarnings() {
181      return warnings;
182   }
183
184   /**
185    * Returns the logger associated with this session.
186    * Subclasses can override this method to provide their own logger.
187    *
188    * @return The logger associated with this session.
189    */
190   protected JuneauLogger getLogger() {
191      if (logger == null)
192         logger = JuneauLogger.getLogger(getClass());
193      return logger;
194   }
195
196   /**
197    * Returns the properties defined on this bean context as a simple map for debugging purposes.
198    *
199    * @return A new map containing the properties defined on this context.
200    */
201   @Overrideable
202   public ObjectMap asMap() {
203      return new ObjectMap();
204   }
205
206   @Override /* Object */
207   public String toString() {
208      try {
209         return asMap().toString(JsonSerializer.DEFAULT_LAX_READABLE);
210      } catch (SerializeException e) {
211         return e.getLocalizedMessage();
212      }
213   }
214
215   /**
216    * Perform cleanup on this context object if necessary.
217    *
218    * @return <jk>true</jk> if this method wasn't previously called.
219    * @throws BeanRuntimeException If called more than once, or in debug mode and warnings occurred.
220    */
221   public boolean close() throws BeanRuntimeException {
222      if (closed)
223         return false;
224      closed = true;
225      return true;
226   }
227
228   @Override /* Object */
229   protected void finalize() throws Throwable {
230//    if (! closed)
231//       throw new RuntimeException("Session was not closed.");
232   }
233}
