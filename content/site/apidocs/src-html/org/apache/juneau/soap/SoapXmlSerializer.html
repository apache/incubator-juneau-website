013package org.apache.juneau.soap;
014
015import static org.apache.juneau.soap.SoapXmlSerializerContext.*;
016
017import org.apache.juneau.*;
018import org.apache.juneau.annotation.*;
019import org.apache.juneau.serializer.*;
020import org.apache.juneau.xml.*;
021
022/**
023 * Serializes POJOs to HTTP responses as XML+SOAP.
024 *
025 * <h5 class='section'>Media types:</h5>
026 * <p>
027 * Handles <code>Accept</code> types: <code>text/xml+soap</code>
028 * <p>
029 * Produces <code>Content-Type</code> types: <code>text/xml+soap</code>
030 *
031 * <h5 class='section'>Description:</h5>
032 * <p>
033 * Essentially the same output as {@link XmlDocSerializer}, except wrapped in a standard SOAP envelope.
034 *
035 * <h5 class='section'>Configurable properties:</h5>
036 * <p>
037 * This class has the following properties associated with it:
038 * <ul>
039 *    <li>{@link SoapXmlSerializerContext}
040 *    <li>{@link BeanContext}
041 * </ul>
042 */
043@Produces(value="text/xml+soap",contentType="text/xml")
044public final class SoapXmlSerializer extends XmlSerializer {
045
046   /**
047    * Constructor.
048    * @param propertyStore The property store containing all the settings for this object.
049    */
050   public SoapXmlSerializer(PropertyStore propertyStore) {
051      super(propertyStore);
052   }
053
054
055   //--------------------------------------------------------------------------------
056   // Overridden methods
057   //--------------------------------------------------------------------------------
058
059   @Override /* Serializer */
060   protected void doSerialize(SerializerSession session, Object o) throws Exception {
061      XmlSerializerSession s = (XmlSerializerSession)session;
062      XmlWriter w = s.getWriter();
063      w.append("<?xml")
064         .attr("version", "1.0")
065         .attr("encoding", "UTF-8")
066         .appendln("?>");
067      w.oTag("soap", "Envelope")
068         .attr("xmlns", "soap", s.getProperty(SOAPXML_SOAPAction, "http://www.w3.org/2003/05/soap-envelope"))
069         .appendln(">");
070      w.sTag(1, "soap", "Body").nl();
071      super.serialize(s, o);
072      w.eTag(1, "soap", "Body").nl();
073      w.eTag("soap", "Envelope").nl();
074   }
075
076   @Override /* Serializer */
077   public ObjectMap getResponseHeaders(ObjectMap properties) {
078      return super.getResponseHeaders(properties)
079         .append("SOAPAction", properties.getString(SOAPXML_SOAPAction, "http://www.w3.org/2003/05/soap-envelope"));
080   }
081}
