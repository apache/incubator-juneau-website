028/**
029 * Encapsulates all access to the properties of a bean class (like a souped-up {@link java.beans.BeanInfo}).
030 *
031 * <h5 class='section'>Description:</h5>
032 * <p>
033 * Uses introspection to find all the properties associated with this class.  If the {@link Bean @Bean} annotation
034 *    is present on the class, or the class has a {@link BeanFilter} registered with it in the bean context,
035 *    then that information is used to determine the properties on the class.
036 * Otherwise, the {@code BeanInfo} functionality in Java is used to determine the properties on the class.
037 *
038 * <h6 class='topic'>Bean property ordering</h6>
039 * <p>
040 * The order of the properties are as follows:
041 * <ul class='spaced-list'>
042 *    <li>If {@link Bean @Bean} annotation is specified on class, then the order is the same as the list of properties in the annotation.
043 *    <li>If {@link Bean @Bean} annotation is not specified on the class, then the order is based on the following.
044 *       <ul>
045 *          <li>Public fields (same order as {@code Class.getFields()}).
046 *          <li>Properties returned by {@code BeanInfo.getPropertyDescriptors()}.
047 *          <li>Non-standard getters/setters with {@link BeanProperty @BeanProperty} annotation defined on them.
048 *       </ul>
049 * </ul>
050 * <br>
051 * The order can also be overridden through the use of an {@link BeanFilter}.
052 *
053 * @param <T> The class type that this metadata applies to.
054 */
055public class BeanMeta<T> {
056
057   /** The target class type that this meta object describes. */
058   protected final ClassMeta<T> classMeta;
059
060   /** The target class that this meta object describes. */
061   protected final Class<T> c;
062
063   /** The properties on the target class. */
064   protected final Map<String,BeanPropertyMeta> properties;
065
066   /** The getter properties on the target class. */
067   protected final Map<Method,String> getterProps;
068
069   /** The setter properties on the target class. */
070   protected final Map<Method,String> setterProps;
071
072   /** The bean context that created this metadata object. */
073   protected final BeanContext ctx;
074
075   /** Optional bean filter associated with the target class. */
076   protected final BeanFilter beanFilter;
077
078   /** Type variables implemented by this bean. */
079   protected final Map<Class<?>,Class<?>[]> typeVarImpls;
080
081   /** The constructor for this bean. */
082   protected final Constructor<T> constructor;
083
084   /** For beans with constructors with BeanConstructor annotation, this is the list of constructor arg properties. */
085   protected final String[] constructorArgs;
086
087   private final MetadataMap extMeta;  // Extended metadata
088
089   // Other fields
090   final String typePropertyName;                                      // "_type" property actual name.
091   private final BeanPropertyMeta typeProperty;                        // "_type" mock bean property.
092   final BeanPropertyMeta dynaProperty;                                // "extras" property.
093   private final String dictionaryName;                                // The @Bean.typeName() annotation defined on this bean class.
094   final String notABeanReason;                                        // Readable string explaining why this class wasn't a bean.
095   final BeanRegistry beanRegistry;
096   final boolean sortProperties;

098   /**
099    * Constructor.
100    *
101    * @param classMeta The target class.
102    * @param ctx The bean context that created this object.
103    * @param beanFilter Optional bean filter associated with the target class.  Can be <jk>null</jk>.
104    * @param pNames Explicit list of property names and order of properties.  If <jk>null</jk>, determine automatically.
105    */
106   protected BeanMeta(final ClassMeta<T> classMeta, BeanContext ctx, BeanFilter beanFilter, String[] pNames) {
107      this.classMeta = classMeta;
108      this.ctx = ctx;
109      this.c = classMeta.getInnerClass();
110
111      Builder<T> b = new Builder<T>(classMeta, ctx, beanFilter, pNames);
112      this.notABeanReason = b.init(this);
113
114      this.beanFilter = beanFilter;
115      this.dictionaryName = b.dictionaryName;
116      this.properties = b.properties == null ? null : Collections.unmodifiableMap(b.properties);
117      this.getterProps = Collections.unmodifiableMap(b.getterProps);
118      this.setterProps = Collections.unmodifiableMap(b.setterProps);
119      this.dynaProperty = b.dynaProperty;
120      this.typeVarImpls = b.typeVarImpls == null ? null : Collections.unmodifiableMap(b.typeVarImpls);
121      this.constructor = b.constructor;
122      this.constructorArgs = b.constructorArgs;
123      this.extMeta = b.extMeta;
124      this.beanRegistry = b.beanRegistry;
125      this.typePropertyName = b.typePropertyName;
126      this.typeProperty = new BeanPropertyMeta.Builder(this, typePropertyName, ctx.string(), beanRegistry).build();
127      this.sortProperties = b.sortProperties;
128   }

137      Map<Method,String> setterProps = new HashMap<Method,String>();
138      BeanPropertyMeta dynaProperty;

149      private Builder(ClassMeta<T> classMeta, BeanContext ctx, BeanFilter beanFilter, String[] pNames) {

156      @SuppressWarnings("unchecked")
157      private String init(BeanMeta<T> beanMeta) {
158         Class<?> c = classMeta.getInnerClass();
159
160         try {
161            Visibility
162               conVis = ctx.beanConstructorVisibility,
163               cVis = ctx.beanClassVisibility,
164               mVis = ctx.beanMethodVisibility,
165               fVis = ctx.beanFieldVisibility;
166
167            List<Class<?>> bdClasses = new ArrayList<Class<?>>();
168            if (beanFilter != null && beanFilter.getBeanDictionary() != null)
169               bdClasses.addAll(Arrays.asList(beanFilter.getBeanDictionary()));
170            Bean bean = classMeta.innerClass.getAnnotation(Bean.class);
171            if (bean != null) {
172               if (! bean.typeName().isEmpty())
173                  bdClasses.add(classMeta.innerClass);
174            }
175            this.beanRegistry = new BeanRegistry(ctx, null, bdClasses.toArray(new Class<?>[bdClasses.size()]));
176
177            for (Bean b : ReflectionUtils.findAnnotationsParentFirst(Bean.class, classMeta.innerClass))
178               if (! b.typePropertyName().isEmpty())
179                  typePropertyName = b.typePropertyName();
180            if (typePropertyName == null)
181               typePropertyName = ctx.getBeanTypePropertyName();
182
183            // If @Bean.interfaceClass is specified on the parent class, then we want
184            // to use the properties defined on that class, not the subclass.
185            Class<?> c2 = (beanFilter != null && beanFilter.getInterfaceClass() != null ? beanFilter.getInterfaceClass() : c);
186
187            Class<?> stopClass = (beanFilter != null ? beanFilter.getStopClass() : Object.class);
188            if (stopClass == null)
189               stopClass = Object.class;
190
191            Map<String,BeanPropertyMeta.Builder> normalProps = new LinkedHashMap<String,BeanPropertyMeta.Builder>();
192
193            /// See if this class matches one the patterns in the exclude-class list.
194            if (ctx.isNotABean(c))
195               return "Class matches exclude-class list";
196
197            if (! (cVis.isVisible(c.getModifiers()) || c.isAnonymousClass()))
198               return "Class is not public";
199
200            if (c.isAnnotationPresent(BeanIgnore.class))
201               return "Class is annotated with @BeanIgnore";
202
203            // Make sure it's serializable.
204            if (beanFilter == null && ctx.beansRequireSerializable && ! isParentClass(Serializable.class, c))
205               return "Class is not serializable";
206
207            // Look for @BeanConstructor constructor.
208            for (Constructor<?> x : c.getConstructors()) {
209               if (x.isAnnotationPresent(BeanConstructor.class)) {
210                  if (constructor != null)
211                     throw new BeanRuntimeException(c, "Multiple instances of '@BeanConstructor' found.");
212                  constructor = (Constructor<T>)x;
213                  constructorArgs = StringUtils.split(x.getAnnotation(BeanConstructor.class).properties(), ',');
214                  if (constructorArgs.length != x.getParameterTypes().length)
215                     throw new BeanRuntimeException(c, "Number of properties defined in '@BeanConstructor' annotation does not match number of parameters in constructor.");
216                  if (! setAccessible(constructor))
217                     throw new BeanRuntimeException(c, "Could not set accessibility to true on method with @BeanConstructor annotation.  Method=''{0}''", constructor.getName());
218               }
219            }
220
221            // If this is an interface, look for impl classes defined in the context.
222            if (constructor == null)
223               constructor = (Constructor<T>)ctx.getImplClassConstructor(c, conVis);
224
225            if (constructor == null)
226               constructor = (Constructor<T>)findNoArgConstructor(c, conVis);
227
228            if (constructor == null && beanFilter == null && ctx.beansRequireDefaultConstructor)
229               return "Class does not have the required no-arg constructor";
230
231            if (! setAccessible(constructor))
232               throw new BeanRuntimeException(c, "Could not set accessibility to true on no-arg constructor");
233
234            // Explicitly defined property names in @Bean annotation.
235            Set<String> fixedBeanProps = new LinkedHashSet<String>();
236
237            if (beanFilter != null) {
238
239               // Get the 'properties' attribute if specified.
240               if (beanFilter.getProperties() != null)
241                  for (String p : beanFilter.getProperties())
242                     fixedBeanProps.add(p);
243
244               if (beanFilter.getPropertyNamer() != null)
245                  propertyNamer = beanFilter.getPropertyNamer();
246            }
247
248            if (propertyNamer == null)
249               propertyNamer = new PropertyNamerDefault();
250
251            // First populate the properties with those specified in the bean annotation to
252            // ensure that ordering first.
253            for (String name : fixedBeanProps)
254               normalProps.put(name, new BeanPropertyMeta.Builder(beanMeta, name));
255
256            if (ctx.useJavaBeanIntrospector) {
257               BeanInfo bi = null;
258               if (! c2.isInterface())
259                  bi = Introspector.getBeanInfo(c2, stopClass);
260               else
261                  bi = Introspector.getBeanInfo(c2, null);
262               if (bi != null) {
263                  for (PropertyDescriptor pd : bi.getPropertyDescriptors()) {
264                     String name = pd.getName();
265                     if (! normalProps.containsKey(name))
266                        normalProps.put(name, new BeanPropertyMeta.Builder(beanMeta, name));
267                     normalProps.get(name).setGetter(pd.getReadMethod()).setSetter(pd.getWriteMethod());
268                  }
269               }
270
271            } else /* Use 'better' introspection */ {
272
273               for (Field f : findBeanFields(c2, stopClass, fVis)) {
274                  String name = findPropertyName(f, fixedBeanProps);
275                  if (name != null) {
276                     if (! normalProps.containsKey(name))
277                        normalProps.put(name, new BeanPropertyMeta.Builder(beanMeta, name));
278                     normalProps.get(name).setField(f);
279                  }
280               }
281
282               List<BeanMethod> bms = findBeanMethods(c2, stopClass, mVis, fixedBeanProps, propertyNamer);
283
284               // Iterate through all the getters.
285               for (BeanMethod bm : bms) {
286                  String pn = bm.propertyName;
287                  Method m = bm.method;
288                  if (! normalProps.containsKey(pn))
289                     normalProps.put(pn, new BeanPropertyMeta.Builder(beanMeta, pn));
290                  BeanPropertyMeta.Builder bpm = normalProps.get(pn);
291                  if (! bm.isSetter)
292                     bpm.setGetter(m);
293               }
294
295               // Now iterate through all the setters.
296               for (BeanMethod bm : bms) {
297                  if (bm.isSetter) {
298                     BeanPropertyMeta.Builder bpm = normalProps.get(bm.propertyName);
299                     if (bm.matchesPropertyType(bpm))
300                        bpm.setSetter(bm.method);
301                  }
302               }
303            }
304
305            typeVarImpls = new HashMap<Class<?>,Class<?>[]>();
306            findTypeVarImpls(c, typeVarImpls);
307            if (typeVarImpls.isEmpty())
308               typeVarImpls = null;
309
310            // Eliminate invalid properties, and set the contents of getterProps and setterProps.
311            for (Iterator<BeanPropertyMeta.Builder> i = normalProps.values().iterator(); i.hasNext();) {
312               BeanPropertyMeta.Builder p = i.next();
313               try {
314                  if (p.validate(ctx, beanRegistry, typeVarImpls)) {
315
316                     if (p.getter != null)
317                        getterProps.put(p.getter, p.name);
318
319                     if (p.setter != null)
320                        setterProps.put(p.setter, p.name);
321
322                  } else {
323                     i.remove();
324                  }
325               } catch (ClassNotFoundException e) {
326                  throw new BeanRuntimeException(c, e.getLocalizedMessage());
327               }
328            }
329
330            // Check for missing properties.
331            for (String fp : fixedBeanProps)
332               if (! normalProps.containsKey(fp))
333                  throw new BeanRuntimeException(c, "The property ''{0}'' was defined on the @Bean(properties=X) annotation but was not found on the class definition.", fp);
334
335            // Mark constructor arg properties.
336            for (String fp : constructorArgs) {
337               BeanPropertyMeta.Builder m = normalProps.get(fp);
338               if (m == null)
339                  throw new BeanRuntimeException(c, "The property ''{0}'' was defined on the @BeanConstructor(properties=X) annotation but was not found on the class definition.", fp);
340               m.setAsConstructorArg();
341            }
342
343            // Make sure at least one property was found.
344            if (beanFilter == null && ctx.beansRequireSomeProperties && normalProps.size() == 0)
345               return "No properties detected on bean class";
346
347            sortProperties = (ctx.sortProperties || (beanFilter != null && beanFilter.isSortProperties())) && fixedBeanProps.isEmpty();
348
349            properties = sortProperties ? new TreeMap<String,BeanPropertyMeta>() : new LinkedHashMap<String,BeanPropertyMeta>();
350
351            if (beanFilter != null && beanFilter.getTypeName() != null)
352               dictionaryName = beanFilter.getTypeName();
353            if (dictionaryName == null)
354               dictionaryName = findDictionaryName(this.classMeta);
355
356            for (Map.Entry<String,BeanPropertyMeta.Builder> e : normalProps.entrySet()) {
357               BeanPropertyMeta pMeta = e.getValue().build();
358               if (pMeta.isDyna())
359                  dynaProperty = pMeta;
360               properties.put(e.getKey(), pMeta);
361            }
362
363            // If a beanFilter is defined, look for inclusion and exclusion lists.
364            if (beanFilter != null) {
365
366               // Eliminated excluded properties if BeanFilter.excludeKeys is specified.
367               String[] includeKeys = beanFilter.getProperties();
368               String[] excludeKeys = beanFilter.getExcludeProperties();
369               if (excludeKeys != null) {
370                  for (String k : excludeKeys)
371                     properties.remove(k);
372
373               // Only include specified properties if BeanFilter.includeKeys is specified.
374               // Note that the order must match includeKeys.
375               } else if (includeKeys != null) {
376                  Map<String,BeanPropertyMeta> properties2 = new LinkedHashMap<String,BeanPropertyMeta>();
377                  for (String k : includeKeys) {
378                     if (properties.containsKey(k))
379                        properties2.put(k, properties.get(k));
380                  }
381                  properties = properties2;
382               }
383            }
384
385            if (pNames != null) {
386               Map<String,BeanPropertyMeta> properties2 = new LinkedHashMap<String,BeanPropertyMeta>();
387               for (String k : pNames) {
388                  if (properties.containsKey(k))
389                     properties2.put(k, properties.get(k));
390               }
391               properties = properties2;
392            }
393
394         } catch (BeanRuntimeException e) {
395            throw e;
396         } catch (Exception e) {
397            return "Exception:  " + StringUtils.getStackTrace(e);
398         }
399
400         return null;
401      }

403      private String findDictionaryName(ClassMeta<?> cm) {
404         BeanRegistry br = cm.getBeanRegistry();
405         if (br != null) {
406            String s = br.getTypeName(this.classMeta);
407            if (s != null)
408               return s;
409         }
410         Class<?> pcm = cm.innerClass.getSuperclass();
411         if (pcm != null) {
412            String s = findDictionaryName(ctx.getClassMeta(pcm));
413            if (s != null)
414               return s;
415         }
416         for (Class<?> icm : cm.innerClass.getInterfaces()) {
417            String s = findDictionaryName(ctx.getClassMeta(icm));
418            if (s != null)
419               return s;
420         }
421         return null;
422      }

424      /*
425       * Returns the property name of the specified field if it's a valid property.
426       * Returns null if the field isn't a valid property.
427       */
428      private String findPropertyName(Field f, Set<String> fixedBeanProps) {
429         BeanProperty bp = f.getAnnotation(BeanProperty.class);
430         if (bp != null && ! bp.name().equals("")) {
431            String name = bp.name();
432            if (fixedBeanProps.isEmpty() || fixedBeanProps.contains(name))
433               return name;
434            throw new BeanRuntimeException(classMeta.getInnerClass(), "Method property ''{0}'' identified in @BeanProperty, but missing from @Bean", name);
435         }
436         String name = propertyNamer.getPropertyName(f.getName());
437         if (fixedBeanProps.isEmpty() || fixedBeanProps.contains(name))
438            return name;
439         return null;
440      }

472   /*
473    * Temporary getter/setter method struct.
474    */
475   private static class BeanMethod {
476      String propertyName;
477      boolean isSetter;
478      Method method;
479      Class<?> type;
480
481      BeanMethod(String propertyName, boolean isSetter, Method method) {
482         this.propertyName = propertyName;
483         this.isSetter = isSetter;
484         this.method = method;
485         if (isSetter)
486            this.type = method.getParameterTypes()[0];
487         else
488            this.type = method.getReturnType();
489      }
490
491      /*
492       * Returns true if this method matches the class type of the specified property.
493       * Only meant to be used for setters.
494       */
495      boolean matchesPropertyType(BeanPropertyMeta.Builder b) {
496         if (b == null)
497            return false;
498
499         // Don't do further validation if this is the "*" bean property.
500         if ("*".equals(b.name))
501            return true;
502
503         // Get the bean property type from the getter/field.
504         Class<?> pt = null;
505         if (b.getter != null)
506            pt = b.getter.getReturnType();
507         else if (b.field != null)
508            pt = b.field.getType();
509
510         // Doesn't match if no getter/field defined.
511         if (pt == null)
512            return false;
513
514         // Doesn't match if not same type or super type as getter/field.
515         if (! isParentClass(type, pt))
516            return false;
517
518         // If a setter was previously set, only use this setter if it's a closer
519         // match (e.g. prev type is a superclass of this type).
520         if (b.setter == null)
521            return true;
522
523         Class<?> prevType = b.setter.getParameterTypes()[0];
524         return isParentClass(prevType, type, true);
525      }
526
527      @Override /* Object */
528      public String toString() {
529         return method.toString();
530      }

533   /*
534    * Find all the bean methods on this class.
535    *
536    * @param c The transformed class.
537    * @param stopClass Don't look above this class in the hierarchy.
538    * @param v The minimum method visibility.
539    * @param fixedBeanProps Only include methods whose properties are in this list.
540    * @param pn Use this property namer to determine property names from the method names.
541    */
542   private static List<BeanMethod> findBeanMethods(Class<?> c, Class<?> stopClass, Visibility v, Set<String> fixedBeanProps, PropertyNamer pn) {
543      List<BeanMethod> l = new LinkedList<BeanMethod>();
544
545      for (Class<?> c2 : findClasses(c, stopClass)) {
546         for (Method m : c2.getDeclaredMethods()) {
547            int mod = m.getModifiers();
548            if (Modifier.isStatic(mod))
549               continue;
550            if (m.isAnnotationPresent(BeanIgnore.class))
551               continue;
552            if (m.isBridge())   // This eliminates methods with covariant return types from parent classes on child classes.
553               continue;
554            if (! (v.isVisible(m) || m.isAnnotationPresent(BeanProperty.class)))
555               continue;
556            String n = m.getName();
557            Class<?>[] pt = m.getParameterTypes();
558            Class<?> rt = m.getReturnType();
559            boolean isGetter = false, isSetter = false;
560            BeanProperty bp = getMethodAnnotation(BeanProperty.class, m);
561            String bpName = bp == null ? "" : bp.name();
562            if (pt.length == 0) {
563               if (n.startsWith("get") && (! rt.equals(Void.TYPE))) {
564                  isGetter = true;
565                  n = n.substring(3);
566               } else if (n.startsWith("is") && (rt.equals(Boolean.TYPE) || rt.equals(Boolean.class))) {
567                  isGetter = true;
568                  n = n.substring(2);
569               } else if (! bpName.isEmpty()) {
570                  isGetter = true;
571                  n = bpName;
572               }
573            } else if (pt.length == 1) {
574               if (n.startsWith("set") && (isParentClass(rt, c) || rt.equals(Void.TYPE))) {
575                  isSetter = true;
576                  n = n.substring(3);
577               } else if (! bpName.isEmpty()) {
578                  isSetter = true;
579                  n = bpName;
580               }
581            } else if (pt.length == 2) {
582               if ("*".equals(bpName)) {
583                  isSetter = true;
584                  n = bpName;
585               }
586            }
587            n = pn.getPropertyName(n);
588            if (isGetter || isSetter) {
589               if (! bpName.isEmpty()) {
590                  n = bpName;
591                  if (! fixedBeanProps.isEmpty())
592                     if (! fixedBeanProps.contains(n))
593                        throw new BeanRuntimeException(c, "Method property ''{0}'' identified in @BeanProperty, but missing from @Bean", n);
594               }
595               l.add(new BeanMethod(n, isSetter, m));
596            }
597         }
598      }
599      return l;
600   }

602   private static Collection<Field> findBeanFields(Class<?> c, Class<?> stopClass, Visibility v) {
603      List<Field> l = new LinkedList<Field>();
604      for (Class<?> c2 : findClasses(c, stopClass)) {
605         for (Field f : c2.getDeclaredFields()) {
606            int m = f.getModifiers();
607            if (Modifier.isStatic(m) || Modifier.isTransient(m))
608               continue;
609            if (f.isAnnotationPresent(BeanIgnore.class))
610               continue;
611            if (! (v.isVisible(f) || f.isAnnotationPresent(BeanProperty.class)))
612               continue;
613            l.add(f);
614         }
615      }
616      return l;
617   }
618
619   private static List<Class<?>> findClasses(Class<?> c, Class<?> stopClass) {
620      LinkedList<Class<?>> l = new LinkedList<Class<?>>();
621      findClasses(c, l, stopClass);
622      return l;
623   }
624
625   private static void findClasses(Class<?> c, LinkedList<Class<?>> l, Class<?> stopClass) {
626      while (c != null && stopClass != c) {
627         l.addFirst(c);
628         for (Class<?> ci : c.getInterfaces())
629            findClasses(ci, l, stopClass);
630         c = c.getSuperclass();
631      }
632   }

711   /**
712    * Recursively determines the classes represented by parameterized types in the class hierarchy of
713    * the specified type, and puts the results in the specified map.<br>
714    * <p>
715    * For example, given the following classes...
716    * <p class='bcode'>
717    *    public static class BeanA&lt;T> {
718    *       public T x;
719    *    }
720    *    public static class BeanB extends BeanA&lt;Integer>} {...}
721    * <p>
722    *    ...calling this method on {@code BeanB.class} will load the following data into {@code m} indicating
723    *    that the {@code T} parameter on the BeanA class is implemented with an {@code Integer}:
724    * <p class='bcode'>
725    *    {BeanA.class:[Integer.class]}
726    * <p>
727    * TODO:  This code doesn't currently properly handle the following situation:
728    * <p class='bcode'>
729    *    public static class BeanB&ltT extends Number> extends BeanA&ltT>;
730    *    public static class BeanC extends BeanB&ltInteger>;
731    * <p>
732    * When called on {@code BeanC}, the variable will be detected as a {@code Number}, not an {@code Integer}.<br>
733    * If anyone can figure out a better way of doing this, please do so!
734    *
735    * @param t The type we're recursing.
736    * @param m Where the results are loaded.
737    */
738   private static void findTypeVarImpls(Type t, Map<Class<?>,Class<?>[]> m) {
739      if (t instanceof Class) {
740         Class<?> c = (Class<?>)t;
741         findTypeVarImpls(c.getGenericSuperclass(), m);
742         for (Type ci : c.getGenericInterfaces())
743            findTypeVarImpls(ci, m);
744      } else if (t instanceof ParameterizedType) {
745         ParameterizedType pt = (ParameterizedType)t;
746         Type rt = pt.getRawType();
747         if (rt instanceof Class) {
748            Type[] gImpls = pt.getActualTypeArguments();
749            Class<?>[] gTypes = new Class[gImpls.length];
750            for (int i = 0; i < gImpls.length; i++) {
751               Type gt = gImpls[i];
752               if (gt instanceof Class)
753                  gTypes[i] = (Class<?>)gt;
754               else if (gt instanceof TypeVariable) {
755                  TypeVariable<?> tv = (TypeVariable<?>)gt;
756                  for (Type upperBound : tv.getBounds())
757                     if (upperBound instanceof Class)
758                        gTypes[i] = (Class<?>)upperBound;
759               }
760            }
761            m.put((Class<?>)rt, gTypes);
762            findTypeVarImpls(pt.getRawType(), m);
763         }
764      }
765   }
