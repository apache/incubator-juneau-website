013package org.apache.juneau.remoteable;
014
015import static org.apache.juneau.internal.StringUtils.*;
016
017import java.lang.annotation.*;
018import java.lang.reflect.*;
019import java.util.*;
020
021import org.apache.juneau.internal.*;
022
023/**
024 * Contains the meta-data about a Java method on a remoteable interface.
025 * <p>
026 * Captures the information in {@link RemoteMethod @RemoteMethod} annotations for caching and reuse.
027 */
028public class RemoteableMethodMeta {
029
030   private final String httpMethod;
031   private final String url;
032   private final RemoteMethodArg[] pathArgs, queryArgs, headerArgs, formDataArgs;
033   private final Integer[] requestBeanArgs, otherArgs;
034   private final Integer bodyArg;
035
036   /**
037    * Constructor.
038    *
039    * @param restUrl The absolute URL of the REST interface backing the interface proxy.
040    * @param m The Java method.
041    */
042   public RemoteableMethodMeta(final String restUrl, Method m) {
043      Builder b = new Builder(restUrl, m);
044      this.httpMethod = b.httpMethod;
045      this.url = b.url;
046      this.pathArgs = b.pathArgs.toArray(new RemoteMethodArg[b.pathArgs.size()]);
047      this.queryArgs = b.queryArgs.toArray(new RemoteMethodArg[b.queryArgs.size()]);
048      this.formDataArgs = b.formDataArgs.toArray(new RemoteMethodArg[b.formDataArgs.size()]);
049      this.headerArgs = b.headerArgs.toArray(new RemoteMethodArg[b.headerArgs.size()]);
050      this.requestBeanArgs = b.requestBeanArgs.toArray(new Integer[b.requestBeanArgs.size()]);
051      this.otherArgs = b.otherArgs.toArray(new Integer[b.otherArgs.size()]);
052      this.bodyArg = b.bodyArg;
053   }
054
055   private static class Builder {
056      private String httpMethod, url;
057      private List<RemoteMethodArg>
058         pathArgs = new LinkedList<RemoteMethodArg>(),
059         queryArgs = new LinkedList<RemoteMethodArg>(),
060         headerArgs = new LinkedList<RemoteMethodArg>(),
061         formDataArgs = new LinkedList<RemoteMethodArg>();
062      private List<Integer>
063         requestBeanArgs = new LinkedList<Integer>(),
064         otherArgs = new LinkedList<Integer>();
065      private Integer bodyArg;
066
067      private Builder(String restUrl, Method m) {
068         Remoteable r = m.getDeclaringClass().getAnnotation(Remoteable.class);
069         RemoteMethod rm = m.getAnnotation(RemoteMethod.class);
070
071         httpMethod = rm == null ? "POST" : rm.httpMethod();
072         if (! isOneOf(httpMethod, "GET", "POST"))
073            throw new RemoteableMetadataException(m, "Invalid value specified for @RemoteMethod.httpMethod() annotation.  Valid values are [GET,POST].");
074
075         String path = rm == null || rm.path().isEmpty() ? null : rm.path();
076         String methodPaths = r == null ? "NAME" : r.methodPaths();
077
078         if (! isOneOf(methodPaths, "NAME", "SIGNATURE"))
079            throw new RemoteableMetadataException(m, "Invalid value specified for @Remoteable.methodPaths() annotation.  Valid values are [NAME,SIGNATURE].");
080
081         url =
082            trimSlashes(restUrl)
083            + '/'
084            + (path != null ? trimSlashes(path) : urlEncode("NAME".equals(methodPaths) ? m.getName() : ClassUtils.getMethodSignature(m)));
085
086         int index = 0;
087         for (Annotation[] aa : m.getParameterAnnotations()) {
088            boolean annotated = false;
089            for (Annotation a : aa) {
090               Class<?> ca = a.annotationType();
091               if (ca == Path.class) {
092                  Path p = (Path)a;
093                  annotated = pathArgs.add(new RemoteMethodArg(p.value(), index, false, p.serializer()));
094               } else if (ca == Query.class) {
095                  Query q = (Query)a;
096                  annotated = queryArgs.add(new RemoteMethodArg(q.value(), index, false, q.serializer()));
097               } else if (ca == QueryIfNE.class) {
098                  QueryIfNE q = (QueryIfNE)a;
099                  annotated = queryArgs.add(new RemoteMethodArg(q.value(), index, true, q.serializer()));
100               } else if (ca == FormData.class) {
101                  FormData f = (FormData)a;
102                  annotated = formDataArgs.add(new RemoteMethodArg(f.value(), index, false, f.serializer()));
103               } else if (ca == FormDataIfNE.class) {
104                  FormDataIfNE f = (FormDataIfNE)a;
105                  annotated = formDataArgs.add(new RemoteMethodArg(f.value(), index, true, f.serializer()));
106               } else if (ca == Header.class) {
107                  Header h = (Header)a;
108                  annotated = headerArgs.add(new RemoteMethodArg(h.value(), index, false, h.serializer()));
109               } else if (ca == HeaderIfNE.class) {
110                  HeaderIfNE h = (HeaderIfNE)a;
111                  annotated = headerArgs.add(new RemoteMethodArg(h.value(), index, true, h.serializer()));
112               } else if (ca == RequestBean.class) {
113                  annotated = true;
114                  requestBeanArgs.add(index);
115               } else if (ca == Body.class) {
116                  annotated = true;
117                  if (bodyArg == null)
118                     bodyArg = index;
119                  else
120                     throw new RemoteableMetadataException(m, "Multiple @Body parameters found.  Only one can be specified per Java method.");
121               }
122            }
123            if (! annotated)
124               otherArgs.add(index);
125            index++;
126         }
127
128         if (bodyArg != null && otherArgs.size() > 0)
129            throw new RemoteableMetadataException(m, "@Body and non-annotated parameters found together.  Non-annotated parameters cannot be used when @Body is used.");
130      }
131   }
132
133   /**
134    * Returns the value of the {@link RemoteMethod#httpMethod()} annotation on this Java method.
135    * @return The value of the {@link RemoteMethod#httpMethod()} annotation, never <jk>null</jk>.
136    */
137   public String getHttpMethod() {
138      return httpMethod;
139   }
140
141   /**
142    * Returns the absolute URL of the REST interface invoked by this Java method.
143    * @return The absolute URL of the REST interface, never <jk>null</jk>.
144    */
145   public String getUrl() {
146      return url;
147   }
148
149   /**
150    * Returns the {@link Path @Path} annotated arguments on this Java method.
151    * @return A map of {@link Path#value() @Path.value()} names to zero-indexed argument indices.
152    */
153   public RemoteMethodArg[] getPathArgs() {
154      return pathArgs;
155   }
156
157   /**
158    * Returns the {@link Query @Query} annotated arguments on this Java method.
159    * @return A map of {@link Query#value() @Query.value()} names to zero-indexed argument indices.
160    */
161   public RemoteMethodArg[] getQueryArgs() {
162      return queryArgs;
163   }
164
165   /**
166    * Returns the {@link FormData @FormData} annotated arguments on this Java method.
167    * @return A map of {@link FormData#value() @FormData.value()} names to zero-indexed argument indices.
168    */
169   public RemoteMethodArg[] getFormDataArgs() {
170      return formDataArgs;
171   }
172
173   /**
174    * Returns the {@link Header @Header} annotated arguments on this Java method.
175    * @return A map of {@link Header#value() @Header.value()} names to zero-indexed argument indices.
176    */
177   public RemoteMethodArg[] getHeaderArgs() {
178      return headerArgs;
179   }
180
181   /**
182    * Returns the {@link RequestBean @RequestBean} annotated arguments on this Java method.
183    * @return A list of zero-indexed argument indices.
184    */
185   public Integer[] getRequestBeanArgs() {
186      return requestBeanArgs;
187   }
188
189   /**
190    * Returns the remaining non-annotated arguments on this Java method.
191    * @return A list of zero-indexed argument indices.
192    */
193   public Integer[] getOtherArgs() {
194      return otherArgs;
195   }
196
197   /**
198    * Returns the argument annotated with {@link Body @Body}.
199    * @return A index of the argument with the {@link Body @Body} annotation, or <jk>null</jk> if no argument exists.
200    */
201   public Integer getBodyArg() {
202      return bodyArg;
203   }
