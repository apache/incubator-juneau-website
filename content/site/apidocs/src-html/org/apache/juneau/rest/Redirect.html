013package org.apache.juneau.rest;
014
015import java.net.*;
016import java.text.*;
017
018import org.apache.juneau.serializer.*;
019import org.apache.juneau.urlencoding.*;
020
021/**
022 * REST methods can return this object as a shortcut for performing <code>HTTP 302</code> redirects.
023 * <p>
024 * The following example shows the difference between handling redirects via the {@link RestRequest}/{@link RestResponse},
025 *    and the simplified approach of using this class.
026 * <p class='bcode'>
027 *    <jc>// Redirect to "/contextPath/servletPath/foobar"</jc>
028 *
029 *    <jc>// Using RestRequest and RestResponse</jc>
030 *    <ja>@RestMethod</ja>(name=<js>"GET"</js>, path=<js>"/example1"</js>)
031 *    <jk>public void</jk> example1(RestRequest req, RestResponse res) <jk>throws</jk> IOException {
032 *       res.sendRedirect(req.getServletURI() + <js>"/foobar"</js>);
033 *    }
034 *
035 *    <jc>// Using Redirect</jc>
036 *    <ja>@RestMethod</ja>(name=<js>"GET"</js>, path=<js>"/example2"</js>)
037 *    <jk>public</jk> Redirect example2() {
038 *       <jk>return new</jk> Redirect(<js>"foobar"</js>);
039 *    }
040 * </p>
041 * <p>
042 * The constructor can use a {@link MessageFormat}-style pattern with multiple arguments:
043 * <p class='bcode'>
044 *    <ja>@RestMethod</ja>(name=<js>"GET"</js>, path=<js>"/example3"</js>)
045 *    <jk>public</jk> Redirect example3() {
046 *       <jk>return new</jk> Redirect(<js>"foo/{0}/bar/{1}"</js>, id1, id2);
047 *    }
048 * </p>
049 * <p>
050 * The arguments are serialized to strings using the servlet's {@link UrlEncodingSerializer},
051 *    so any filters defined on the serializer or REST method/class will be used when present.
052 * The arguments will also be automatically URL-encoded.
053 * <p>
054 * Redirecting to the servlet root can be accomplished by simply using the no-arg constructor.
055 * <p class='bcode'>
056 *    <jc>// Simply redirect to the servlet root.
057 *    // Equivalent to res.sendRedirect(req.getServletURI()).</jc>
058 *    <ja>@RestMethod</ja>(name=<js>"GET"</js>, path=<js>"/example4"</js>)
059 *    <jk>public</jk> Redirect exmaple4() {
060 *       <jk>return new</jk> Redirect();
061 *    }
062 * </p>
063 * <p>
064 * This class is handled by {@link org.apache.juneau.rest.response.RedirectHandler}, a built-in default
065 *    response handler created in {@link RestConfig}.
066 */
067public final class Redirect {
068
069   private final int httpResponseCode;
070   private final String url;
071   private final Object[] args;
072
073   /**
074    * Redirect to the specified URL.
075    * Relative paths are interpreted as relative to the servlet path.
076    *
077    * @param url The URL to redirect to.
078    * <br>Can be any of the following:
079    * <ul>
080    *    <li><code>URL</code>
081    *    <li><code>URI</code>
082    *    <li><code>CharSequence</code>
083    * </ul>
084    * @param args Optional {@link MessageFormat}-style arguments.
085    */
086   public Redirect(Object url, Object...args) {
087      this(0, url, args);
088   }
089
090   /**
091    * Convenience method for redirecting to instance of {@link URL} and {@link URI}.
092    * Same as calling <code>toString()</code> on the object and using the other constructor.
093    *
094    * @param url The URL to redirect to.
095    * <br>Can be any of the following:
096    * <ul>
097    *    <li><code>URL</code>
098    *    <li><code>URI</code>
099    *    <li><code>CharSequence</code>
100    * </ul>
101    */
102   public Redirect(Object url) {
103      this(0, url, (Object[])null);
104   }
105
106   /**
107    * Redirect to the specified URL.
108    * Relative paths are interpreted as relative to the servlet path.
109    *
110    * @param httpResponseCode The HTTP response code.
111    * @param url The URL to redirect to.
112    * <br>Can be any of the following:
113    * <ul>
114    *    <li><code>URL</code>
115    *    <li><code>URI</code>
116    *    <li><code>CharSequence</code>
117    * </ul>
118    * @param args Optional {@link MessageFormat}-style arguments.
119    */
120   public Redirect(int httpResponseCode, Object url, Object...args) {
121      this.httpResponseCode = httpResponseCode;
122      this.url = (url == null ? null : url.toString());
123      this.args = args;
124   }
125
126   /**
127    * Shortcut for redirecting to the servlet root.
128    */
129   public Redirect() {
130      this(0, null, (Object[])null);
131   }
132
133   /**
134    * Calculates the URL to redirect to.
135    *
136    * @param s Use this serializer to encode arguments using the {@link UrlEncodingSerializer#serialize(PartType,Object)} method.
137    * @return The URL to redirect to.
138    */
139   public String toUrl(UrlEncodingSerializer s) {
140      if (url != null && args != null && args.length > 0) {
141         for (int i = 0; i < args.length; i++)
142            args[i] = s.serialize(PartType.PATH, args[i]);
143         return MessageFormat.format(url, args);
144      }
145      return url;
146   }
147
148   /**
149    * Returns the response code passed in through the constructor.
150    *
151    * @return The response code passed in through the constructor, or <code>0</code> if response code wasn't specified.
152    */
153   public int getHttpResponseCode() {
154      return httpResponseCode;
155   }
156}
