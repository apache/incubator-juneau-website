056   /**
057    * Adds default entries to these headers.
058    * <p>
059    * This includes the default headers defined on the servlet and method levels.
060    *
061    * @param defaultEntries The default entries.  Can be <jk>null</jk>.
062    * @return This object (for method chaining).
063    */
064   public RequestHeaders addDefault(Map<String,String> defaultEntries) {
065      if (defaultEntries != null) {
066         for (Map.Entry<String,String> e : defaultEntries.entrySet()) {
067            String key = e.getKey(), value = e.getValue();
068            String[] v = get(key);
069            if (v == null || v.length == 0 || StringUtils.isEmpty(v[0]))
070               put(key, new String[]{value});
071         }
072      }
073      return this;
074   }
075
076   /**
077    * Adds a set of header values to this object.
078    *
079    * @param name The header name.
080    * @param values The header values.
081    * @return This object (for method chaining).
082    */
083   public RequestHeaders put(String name, Enumeration<String> values) {
084      // Optimized for enumerations of one entry, the most-common case.
085      if (values.hasMoreElements()) {
086         String v = values.nextElement();
087         String[] s = new String[]{v};
088         while (values.hasMoreElements())
089            s = ArrayUtils.append(s, values.nextElement());
090         put(name, s);
091      }
092      return this;
093   }

095   /**
096    * Returns the specified header value, or <jk>null</jk> if the header doesn't exist.
097    * <p>
098    * If {@code allowHeaderParams} init parameter is <jk>true</jk>, then first looks for {@code &HeaderName=x} in the URL query string.
099    * <p>
100    * @param name The header name.
101    * @return The header value, or <jk>null</jk> if it doesn't exist.
102    */
103   public String getFirst(String name) {
104      String[] v = null;
105      if (queryParams != null)
106         v = queryParams.get(name);
107      if (v == null || v.length == 0)
108         v = get(name);
109      if (v == null || v.length == 0)
110         return null;
111      return v[0];
112   }
113
114   /**
115    * Returns the specified header value, or a default value if the header doesn't exist.
116    * <p>
117    * If {@code allowHeaderParams} init parameter is <jk>true</jk>, then first looks for {@code &HeaderName=x} in the URL query string.
118    *
119    * @param name The HTTP header name.
120    * @param def The default value to return if the header value isn't found.
121    * @return The header value, or the default value if the header isn't present.
122    */
123   public String getFirst(String name, String def) {
124      String s = getFirst(name);
125      return StringUtils.isEmpty(s) ? def : s;
126   }

138   /**
139    * Returns the specified header value converted to a POJO.
140    * <p>
141    * The type can be any POJO type convertable from a <code>String</code>
142    * (See <a class="doclink" href="package-summary.html#PojosConvertableFromString">POJOs Convertable From Strings</a>).
143    * <p>
144    * <h5 class='section'>Examples:</h5>
145    * <p class='bcode'>
146    *    <jc>// Parse into an integer.</jc>
147    *    <jk>int</jk> myheader = req.getHeader(<js>"My-Header"</js>, <jk>int</jk>.<jk>class</jk>);
148    *
149    *    <jc>// Parse a UUID.</jc>
150    *    UUID myheader = req.getHeader(<js>"My-Header"</js>, UUID.<jk>class</jk>);
151    * </p>
152    *
153    * @param name The HTTP header name.
154    * @param type The class type to convert the header value to.
155    * @param <T> The class type to convert the header value to.
156    * @return The parameter value converted to the specified class type.
157    */
158   public <T> T get(String name, Class<T> type) {
159      String h = getFirst(name);
160      return beanSession.convertToType(h, type);
161   }
162
163   /**
164    * Same as {@link #get(String, Class)} but returns a default value if not found.
165    *
166    * @param name The HTTP header name.
167    * @param def The default value if the header was not specified or is <jk>null</jk>.
168    * @param type The class type to convert the header value to.
169    * @param <T> The class type to convert the header value to.
170    * @return The parameter value converted to the specified class type.
171    */
172   public <T> T get(String name, T def, Class<T> type) {
173      String h = getFirst(name);
174      if (h == null)
175         return def;
176      return beanSession.convertToType(h, type);
177   }

179   /**
180    * Returns the specified header value converted to a POJO.
181    * <p>
182    * The type can be any POJO type convertable from a <code>String</code>
183    * (See <a class="doclink" href="package-summary.html#PojosConvertableFromString">POJOs Convertable From Strings</a>).
184    * <p>
185    * <h5 class='section'>Examples:</h5>
186    * <p class='bcode'>
187    *    <jc>// Parse into a linked-list of strings.</jc>
188    *    List&lt;String&gt; myheader = req.getHeader(<js>"My-Header"</js>, LinkedList.<jk>class</jk>, String.<jk>class</jk>);
189    * </p>
190    *
191    * @param name The HTTP header name.
192    * @param type The type of object to create.
193    *    <br>Can be any of the following: {@link ClassMeta}, {@link Class}, {@link ParameterizedType}, {@link GenericArrayType}
194    * @param args The type arguments of the class if it's a collection or map.
195    *    <br>Can be any of the following: {@link ClassMeta}, {@link Class}, {@link ParameterizedType}, {@link GenericArrayType}
196    *    <br>Ignored if the main type is not a map or collection.
197    * @param <T> The class type to convert the header value to.
198    * @return The parameter value converted to the specified class type.
199    * @throws ParseException If the header could not be converted to the specified type.
200    */
201   public <T> T get(String name, Type type, Type...args) throws ParseException {
202      String h = getFirst(name);
203      return parser.parsePart(h, type, args);
204   }
