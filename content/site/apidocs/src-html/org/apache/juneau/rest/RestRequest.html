044/**
045 * Represents an HTTP request for a REST resource.
046 * <p>
047 * Equivalent to {@link HttpServletRequest} except with some additional convenience methods.
048 * <p>
049 * For reference, given the URL <js>"http://localhost:9080/contextRoot/servletPath/foo?bar=baz#qux"</js>, the
050 *    following methods return the following values....
051 * <table class='styled'>
052 *    <tr><th>Method</th><th>Value</th></tr>
053 *    <tr><td>{@code getContextPath()}</td><td>{@code /contextRoot}</td></tr>
054 *    <tr><td>{@code getPathInfo()}</td><td>{@code /foo}</td></tr>
055 *    <tr><td>{@code getPathTranslated()}</td><td>{@code path-to-deployed-war-on-filesystem/foo}</td></tr>
056 *    <tr><td>{@code getQueryString()}</td><td>{@code bar=baz}</td></tr>
057 *    <tr><td>{@code getRequestURI()}</td><td>{@code /contextRoot/servletPath/foo}</td></tr>
058 *    <tr><td>{@code getRequestURL()}</td><td>{@code http://localhost:9080/contextRoot/servletPath/foo}</td></tr>
059 *    <tr><td>{@code getServletPath()}</td><td>{@code /servletPath}</td></tr>
060 * </table>
061 * <p>
062 * Refer to <a class="doclink" href="package-summary.html#TOC">REST Servlet API</a> for information about using this class.
063 */
064@SuppressWarnings("unchecked")

092   /**
093    * Constructor.
094    */
095   RestRequest(RestContext context, HttpServletRequest req) throws ServletException {
096      super(req);
097      this.context = context;
098
099      try {
100         isPost = req.getMethod().equalsIgnoreCase("POST");
101
102         // If this is a POST, we want to parse the query parameters ourselves to prevent
103         // the servlet code from processing the HTTP body as URL-Encoded parameters.
104         queryParams = new RequestQuery();
105         if (isPost)
106            context.getUrlEncodingParser().parseIntoSimpleMap(getQueryString(), queryParams);
107         else
108            queryParams.putAll(req.getParameterMap());
109
110
111         // Get the HTTP method.
112         // Can be overridden through a "method" GET attribute.
113         String _method = super.getMethod();
114
115         String m = getQuery().getFirst("method");
116         if (context.allowMethodParam(m))
117            _method = m;
118
119         method = _method;
120
121         headers = new RequestHeaders();
122         for (Enumeration<String> e = getHeaderNames(); e.hasMoreElements();) {
123            String name = e.nextElement();
124            headers.put(name, super.getHeaders(name));
125         }
126
127         body = new RequestBody(this);
128
129         if (context.isAllowBodyParam()) {
130            String b = getQuery().getFirst("body");
131            if (b != null) {
132               headers.put("Content-Type", UonSerializer.DEFAULT.getResponseContentType());
133               body.load(b.getBytes(IOUtils.UTF8));
134            }
135         }
136
137         if (context.isAllowHeaderParams())
138            headers.setQueryParams(queryParams);
139
140         debug = "true".equals(getQuery().getFirst("debug", "false")) || "true".equals(getHeaders().getFirst("Debug", "false"));
141
142         this.pathParams = new RequestPathMatch();
143
144      } catch (RestException e) {
145         throw e;
146      } catch (Exception e) {
147         throw new ServletException(e);
148      }
149   }

151   /*
152    * Called from RestServlet after a match has been made but before the guard or method invocation.
153    */
154   @SuppressWarnings("hiding")
155   final void init(Method javaMethod, ObjectMap properties, Map<String,String> defHeader,
156         Map<String,String> defQuery, Map<String,String> defFormData, String defaultCharset,
157         SerializerGroup mSerializers, ParserGroup mParsers, UrlEncodingParser mUrlEncodingParser,
158         EncoderGroup encoders, String pageTitle, String pageText, String pageLinks) {
159      this.javaMethod = javaMethod;
160      this.properties = properties;
161      this.urlEncodingParser = mUrlEncodingParser;
162      this.beanSession = urlEncodingParser.getBeanContext().createSession();
163      this.pathParams
164         .setParser(urlEncodingParser)
165         .setBeanSession(beanSession);
166      this.queryParams
167         .addDefault(defQuery)
168         .setParser(urlEncodingParser)
169         .setBeanSession(beanSession);
170      this.headers
171         .addDefault(defHeader)
172         .addDefault(context.getDefaultRequestHeaders())
173         .setParser(urlEncodingParser)
174         .setBeanSession(beanSession);
175      this.body
176         .setEncoders(encoders)
177         .setParsers(mParsers)
178         .setHeaders(headers)
179         .setBeanSession(beanSession)
180         .setUrlEncodingParser(mUrlEncodingParser);
181      this.serializerGroup = mSerializers;
182      this.parserGroup = mParsers;
183      this.defaultCharset = defaultCharset;
184      this.pageTitle = pageTitle;
185      this.pageText = pageText;
186      this.pageLinks = pageLinks;
187      this.defFormData = defFormData;
188
189      if (debug) {
190         String msg = ""
191            + "\n=== HTTP Request (incoming) ===================================================="
192            + toString()
193            + "\n=== END ========================================================================";
194         context.getLogger().log(Level.WARNING, msg);
195      }
196   }

276   /**
277    * Returns the charset specified on the <code>Content-Type</code> header, or <js>"UTF-8"</js> if not specified.
278    */
279   @Override /* ServletRequest */
280   public String getCharacterEncoding() {
281      if (charset == null) {
282         // Determine charset
283         // NOTE:  Don't use super.getCharacterEncoding() because the spec is implemented inconsistently.
284         // Jetty returns the default charset instead of null if the character is not specified on the request.
285         String h = getHeader("Content-Type");
286         if (h != null) {
287            int i = h.indexOf(";charset=");
288            if (i > 0)
289               charset = h.substring(i+9).trim();
290         }
291         if (charset == null)
292            charset = defaultCharset;
293         if (! Charset.isSupported(charset))
294            throw new RestException(SC_UNSUPPORTED_MEDIA_TYPE, "Unsupported charset in header ''Content-Type'': ''{0}''", h);
295      }
296      return charset;
297   }

299   @Override /* ServletRequest */
300   public Locale getLocale() {
301      String h = headers.getFirst("Accept-Language");
302      if (h != null) {
303         MediaTypeRange[] mr = MediaTypeRange.parse(h);
304         if (mr.length > 0)
305            return toLocale(mr[0].getMediaType().getType());
306      }
307      return super.getLocale();
308   }
309
310   @Override /* ServletRequest */
311   public Enumeration<Locale> getLocales() {
312      String h = headers.getFirst("Accept-Language");
313      if (h != null) {
314         MediaTypeRange[] mr = MediaTypeRange.parse(h);
315         if (mr.length > 0) {
316            List<Locale> l = new ArrayList<Locale>(mr.length);
317            for (MediaTypeRange r : mr)
318               l.add(toLocale(r.getMediaType().getType()));
319            return enumeration(l);
320         }
321      }
322      return super.getLocales();
323   }

357   /**
358    * Retrieves the URL-encoded form data from the request if the body has already been cached locally.
359    *
360    * @return The URL-encoded form data from the request.
361    */
362   public RequestFormData getFormData() {
363      try {
364         if (formData == null) {
365            formData = new RequestFormData();
366            formData.setParser(urlEncodingParser).setBeanSession(beanSession);
367            if (! body.isLoaded()) {
368               formData.putAll(getParameterMap());
369            } else {
370               Map<String,String> m = urlEncodingParser.parse(body.getReader(), Map.class, String.class, String.class);
371               for (Map.Entry<String,String> e : m.entrySet()) {
372                  formData.put(e.getKey(), e.getValue());
373               }
374            }
375         }
376         formData.addDefault(defFormData);
377         return formData;
378      } catch (Exception e) {
379         throw new RestException(SC_INTERNAL_SERVER_ERROR, e);
380      }
381   }

480   /**
481    * Returns the value {@link #getPathInfo()} split on the <js>'/'</js> character.
482    * <p>
483    * If path info is <jk>null</jk>, returns an empty list.
484    * <p>
485    * URL-encoded characters in segments are automatically decoded by this method.
486    *
487    * @return The decoded segments, or an empty list if path info is <jk>null</jk>.
488    */
489   public String[] getPathInfoParts() {
490      String s = getPathInfoUndecoded();
491      if (s == null || s.isEmpty() || s.equals("/"))
492         return new String[0];
493      s = s.substring(1);
494      if (s.endsWith("/"))
495         s = s.substring(0, s.length()-1);
496      boolean needsDecode = (s.indexOf('%') != -1 || s.indexOf('+') != -1);
497      String[] l = s.split("/", Integer.MAX_VALUE);
498      try {
499         if (needsDecode)
500            for (int i = 0; i < l.length; i++)
501               l[i] = URLDecoder.decode(l[i], "UTF-8");
502      } catch (UnsupportedEncodingException e) {
503         e.printStackTrace();  // Won't happen.
504      }
505      return l;
506   }

508   /**
509    * Returns a resolved URL.
510    * <p>
511    * <ul class='spaced-list'>
512    *    <li>Fully-qualified absolute URLs (e.g. <js>"http://..."</js>, <js>"https://"</js>) are simply converted to a URL.
513    *    <li>Absolute URLs (e.g. <js>"/foo/..."</js>) are interpreted as relative to the server hostname.
514    *    <li>Relative URLs (e.g. <js>"foo/..."</js>) are interpreted as relative to this servlet path.
515    * </ul>
516    *
517    * @param path The URL path to resolve.
518    * @return The resolved URL.
519    * @throws MalformedURLException If path is not a valid URL component.
520    */
521   public URL getURL(String path) throws MalformedURLException {
522      if (path.startsWith("http://") || path.startsWith("https://"))
523         return new URL(path);
524      if (startsWith(path, '/'))
525         return new URL(getScheme(), getLocalName(), getLocalPort(), path);
526      return new URL(getScheme(), getLocalName(), getLocalPort(), getContextPath() + getServletPath() + (isEmpty(path) ? "" : ('/' + path)));
527   }

539   /**
540    * Returns the URI of the parent resource.
541    * <p>
542    * Trailing slashes in the path are ignored by this method.
543    * <p>
544    * The behavior is shown below:
545    * <table class='styled'>
546    *    <tr>
547    *       <th>getRequestURI</th>
548    *       <th>getRequestParentURI</th>
549    *    </tr>
550    *    <tr>
551    *       <td><code>/foo/bar</code></td>
552    *       <td><code>/foo</code></td>
553    *    </tr>
554    *    <tr>
555    *       <td><code>/foo/bar?baz=bing</code></td>
556    *       <td><code>/foo</code></td>
557    *    </tr>
558    *    <tr>
559    *       <td><code>/foo/bar/</code></td>
560    *       <td><code>/foo</code></td>
561    *    </tr>
562    *    <tr>
563    *       <td><code>/foo/bar//</code></td>
564    *       <td><code>/foo</code></td>
565    *    </tr>
566    *    <tr>
567    *       <td><code>/foo//bar//</code></td>
568    *       <td><code>/foo/</code></td>
569    *    </tr>
570    *    <tr>
571    *       <td><code>/foo</code></td>
572    *       <td>/</td>
573    *    </tr>
574    * </table>
575    *
576    * @return The request parent URI.
577    */
578   public String getRequestParentURI() {
579      String uri = getRequestURI();
580      while (endsWith(uri, '/'))
581         uri = uri.substring(0, uri.length()-1);
582      int i = uri.lastIndexOf('/');
583      if (i <= 0)
584         return "/";
585      return uri.substring(0, i);
586   }

692    *
693    * @return The localized page title for HTML views.

695   protected String getPageTitle() {
696      String s = pageTitle;
697      if (isEmpty(s))
698         s = context.getMessages().findFirstString(getLocale(), javaMethod.getName() + ".pageTitle");
699      if (isEmpty(s))
700         s = context.getMessages().findFirstString(getLocale(), "pageTitle");
701      if (! isEmpty(s))
702         return resolveVars(s);
703      s = getServletTitle();
704      return s;
705   }

713      String s = pageText;

715         s = context.getMessages().findFirstString(getLocale(), javaMethod.getName() + ".pageText");
716      if (isEmpty(s))
717         s = context.getMessages().findFirstString(getLocale(), "pageText");
718      if (! isEmpty(s))
719         return resolveVars(s);
720      s = getMethodSummary();
721      if (isEmpty(s))
722         s = getServletDescription();
723      return s;
724   }

973   @Override /* Object */
974   public String toString() {
975      StringBuilder sb = new StringBuilder("\n").append(getDescription()).append("\n");
976      sb.append("---Headers---\n");
977      for (Enumeration<String> e = getHeaderNames(); e.hasMoreElements();) {
978         String h = e.nextElement();
979         sb.append("\t").append(h).append(": ").append(getHeader(h)).append("\n");
980      }
981      sb.append("---Default Servlet Headers---\n");
982      for (Map.Entry<String,String> e : context.getDefaultRequestHeaders().entrySet()) {
983         sb.append("\t").append(e.getKey()).append(": ").append(e.getValue()).append("\n");
984      }
985      if (javaMethod == null) {
986         sb.append("***init() not called yet!***\n");
987      } else if (method.equals("PUT") || method.equals("POST")) {
988         try {
989            sb.append("---Body UTF-8---\n");
990            sb.append(body.asString()).append("\n");
991            sb.append("---Body Hex---\n");
992            sb.append(body.asHex()).append("\n");
993         } catch (Exception e1) {
994            sb.append(e1.getLocalizedMessage());
995            context.getLogger().log(WARNING, e1, "Error occurred while trying to read debug input.");
996         }
997      }
998      return sb.toString();
999   }

1007    * Converts an Accept-Language value entry to a Locale.
1008    */
1009   private static Locale toLocale(String lang) {
1010      String country = "";
1011      int i = lang.indexOf('-');
1012      if (i > -1) {
1013         country = lang.substring(i+1).trim();
1014         lang = lang.substring(0,i).trim();
1015      }
1016      return new Locale(lang, country);
1017   }
