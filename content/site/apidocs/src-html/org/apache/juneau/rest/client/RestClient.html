133   /**
134    * Calls {@link CloseableHttpClient#close()} on the underlying {@link CloseableHttpClient}.
135    * It's good practice to call this method after the client is no longer used.
136    *
137    * @throws IOException
138    */
139   public void close() throws IOException {
140      isClosed = true;
141      if (httpClient != null && ! keepHttpClientOpen)
142         httpClient.close();
143      if (executorService != null && executorServiceShutdownOnClose)
144         executorService.shutdown();
145      if (Boolean.getBoolean("org.apache.juneau.rest.client.RestClient.trackLifecycle"))
146         closedStack = Thread.currentThread().getStackTrace();
147   }

149   /**
150    * Same as {@link #close()}, but ignores any exceptions.
151    */
152   public void closeQuietly() {
153      isClosed = true;
154      try {
155         if (httpClient != null && ! keepHttpClientOpen)
156            httpClient.close();
157         if (executorService != null && executorServiceShutdownOnClose)
158            executorService.shutdown();
159      } catch (Throwable t) {}
160      if (Boolean.getBoolean("org.apache.juneau.rest.client.RestClient.trackLifecycle"))
161         closedStack = Thread.currentThread().getStackTrace();
162   }

282   /**
283    * Performs a REST call where the entire call is specified in a simple string.
284    * <p>
285    * This method is useful for performing callbacks when the target of a callback is passed in
286    * on an initial request, for example to signal when a long-running process has completed.
287    * <p>
288    * The call string can be any of the following formats:
289    * <ul class='spaced-list'>
290    *    <li><js>"[method] [url]"</js> - e.g. <js>"GET http://localhost/callback"</js>
291    *    <li><js>"[method] [url] [payload]"</js> - e.g. <js>"POST http://localhost/callback some text payload"</js>
292    *    <li><js>"[method] [headers] [url] [payload]"</js> - e.g. <js>"POST {'Content-Type':'text/json'} http://localhost/callback {'some':'json'}"</js>
293    * </ul>
294    * <p>
295    * The payload will always be sent using a simple {@link StringEntity}.
296    *
297    * @param callString The call string.
298    * @return A {@link RestCall} object that can be further tailored before executing the request
299    *    and getting the response as a parsed object.
300    * @throws RestCallException
301    */
302   public RestCall doCallback(String callString) throws RestCallException {
303      String s = callString;
304      try {
305         RestCall rc = null;
306         String method = null, uri = null, content = null;
307         ObjectMap h = null;
308         int i = s.indexOf(' ');
309         if (i != -1) {
310            method = s.substring(0, i).trim();
311            s = s.substring(i).trim();
312            if (s.length() > 0) {
313               if (s.charAt(0) == '{') {
314                  i = s.indexOf('}');
315                  if (i != -1) {
316                     String json = s.substring(0, i+1);
317                     h = JsonParser.DEFAULT.parse(json, ObjectMap.class);
318                     s = s.substring(i+1).trim();
319                  }
320               }
321               if (s.length() > 0) {
322                  i = s.indexOf(' ');
323                  if (i == -1)
324                     uri = s;
325                  else {
326                     uri = s.substring(0, i).trim();
327                     s = s.substring(i).trim();
328                     if (s.length() > 0)
329                        content = s;
330                  }
331               }
332            }
333         }
334         if (method != null && uri != null) {
335            rc = doCall(method, uri, content != null);
336            if (content != null)
337               rc.input(new StringEntity(content));
338            if (h != null)
339               for (Map.Entry<String,Object> e : h.entrySet())
340                  rc.header(e.getKey(), e.getValue());
341            return rc;
342         }
343      } catch (Exception e) {
344         throw new RestCallException(e);
345      }
346      throw new RestCallException("Invalid format for call string.");
347   }

369      RestCall rc = doCall(method.name(), url, method.hasContent());
370      if (method.hasContent())
371         rc.input(content);
372      return rc;
373   }

375   /**
376    * Perform a generic REST call.
377    *
378    * @param method The method name (e.g. <js>"GET"</js>, <js>"OPTIONS"</js>).
379    * @param url The URL of the remote REST resource.  Can be any of the following:  {@link String}, {@link URI}, {@link URL}.
380    * @param hasContent Boolean flag indicating if the specified request has content associated with it.
381    * @return A {@link RestCall} object that can be further tailored before executing the request
382    *    and getting the response as a parsed object.
383    * @throws RestCallException If any authentication errors occurred.
384    */
385   public RestCall doCall(String method, Object url, boolean hasContent) throws RestCallException {
386      if (isClosed) {
387         Exception e2 = null;
388         if (closedStack != null) {
389            e2 = new Exception("Creation stack:");
390            e2.setStackTrace(closedStack);
391            throw new RestCallException("RestClient.close() has already been called.  This client cannot be reused.").initCause(e2);
392         }
393         throw new RestCallException("RestClient.close() has already been called.  This client cannot be reused.  Closed location stack trace can be displayed by setting the system property 'org.apache.juneau.rest.client.RestClient.trackCreation' to true.");
394      }
395
396      HttpRequestBase req = null;
397      RestCall restCall = null;
398      final String methodUC = method.toUpperCase(Locale.ENGLISH);
399      try {
400         if (hasContent) {
401            req = new HttpEntityEnclosingRequestBase() {
402               @Override /* HttpRequest */
403               public String getMethod() {
404                  return methodUC;
405               }
406            };
407            restCall = new RestCall(this, req, toURI(url));
408         } else {
409            req = new HttpRequestBase() {
410               @Override /* HttpRequest */
411               public String getMethod() {
412                  return methodUC;
413               }
414            };
415            restCall = new RestCall(this, req, toURI(url));
416         }
417      } catch (URISyntaxException e1) {
418         throw new RestCallException(e1);
419      }
420      for (Map.Entry<String,? extends Object> e : headers.entrySet())
421         restCall.header(e.getKey(), e.getValue());
422
423      if (parser != null && ! req.containsHeader("Accept"))
424         req.setHeader("Accept", parser.getPrimaryMediaType().toString());
425
426      return restCall;
427   }

429   /**
430    * Create a new proxy interface against a REST interface.
431    * <p>
432    * The URL to the REST interface is based on the following values:
433    * <ul>
434    *    <li>The {@link Remoteable#path() @Remoteable.path()} annotation on the interface (<code>remoteable-path</code>).
435    *    <li>The {@link RestClientBuilder#rootUrl(Object) rootUrl} on the client (<code>root-url</code>).
436    *    <li>The fully-qualified class name of the interface (<code>class-name</code>).
437    * </ul>
438    * <p>
439    * The URL calculation is as follows:
440    * <ul>
441    *    <li><code>remoteable-path</code> - If remoteable path is absolute.
442    *    <li><code>root-url/remoteable-path</code> - If remoteable path is relative and root-url has been specified.
443    *    <li><code>root-url/class-name</code> - If remoteable path is not specified.
444    * </ul>
445    * <p>
446    * If the information is not available to resolve to an absolute URL, a {@link RemoteableMetadataException} is thrown.
447    * <p>
448    * Examples:
449    * <p class='bcode'>
450    *    <jk>package</jk> org.apache.foo;
451    *
452    *    <ja>@Remoteable</ja>(path=<js>"http://hostname/resturl/myinterface1"</js>)
453    *    <jk>public interface</jk> MyInterface1 { ... }
454    *
455    *    <ja>@Remoteable</ja>(path=<js>"/myinterface2"</js>)
456    *    <jk>public interface</jk> MyInterface2 { ... }
457    *
458    *    <jk>public interface</jk> MyInterface3 { ... }
459    *
460    *    <jc>// Resolves to "http://localhost/resturl/myinterface1"</jc>
461    *    MyInterface1 i1 = <jk>new</jk> RestClientBuilder()
462    *       .build()
463    *       .getRemoteableProxy(MyInterface1.<jk>class</jk>);
464    *
465    *    <jc>// Resolves to "http://hostname/resturl/myinterface2"</jc>
466    *    MyInterface2 i2 = <jk>new</jk> RestClientBuilder()
467    *       .rootUrl(<js>"http://hostname/resturl"</js>)
468    *       .build()
469    *       .getRemoteableProxy(MyInterface2.<jk>class</jk>);
470    *
471    *    <jc>// Resolves to "http://hostname/resturl/org.apache.foo.MyInterface3"</jc>
472    *    MyInterface3 i3 = <jk>new</jk> RestClientBuilder()
473    *       .rootUrl(<js>"http://hostname/resturl"</js>)
474    *       .build()
475    *       .getRemoteableProxy(MyInterface3.<jk>class</jk>);
476    * </p>
477    *
478    * @param interfaceClass The interface to create a proxy for.
479    * @return The new proxy interface.
480    * @throws RemoteableMetadataException If the REST URI cannot be determined based on the information given.
481    */
482   public <T> T getRemoteableProxy(final Class<T> interfaceClass) {
483      return getRemoteableProxy(interfaceClass, null);
484   }

497   /**
498    * Same as {@link #getRemoteableProxy(Class, Object)} but allows you to override the serializer and parser used.
499    *
500    * @param interfaceClass The interface to create a proxy for.
501    * @param restUrl The URL of the REST interface.
502    * @param serializer The serializer used to serialize POJOs to the body of the HTTP request.
503    * @param parser The parser used to parse POJOs from the body of the HTTP response.
504    * @return The new proxy interface.
505    */
506   @SuppressWarnings({ "unchecked", "hiding" })
507   public <T> T getRemoteableProxy(final Class<T> interfaceClass, Object restUrl, final Serializer serializer, final Parser parser) {
508
509      if (restUrl == null) {
510         Remoteable r = ReflectionUtils.getAnnotation(Remoteable.class, interfaceClass);
511
512         String path = r == null ? "" : trimSlashes(r.path());
513         if (path.indexOf("://") == -1) {
514            if (path.isEmpty())
515               path = interfaceClass.getName();
516            if (rootUrl == null)
517               throw new RemoteableMetadataException(interfaceClass, "Root URI has not been specified.  Cannot construct absolute path to remoteable proxy.");
518            path = trimSlashes(rootUrl) + '/' + path;
519         }
520         restUrl = path;
521      }
522
523      final String restUrl2 = restUrl.toString();
524
525      try {
526         return (T)Proxy.newProxyInstance(
527            interfaceClass.getClassLoader(),
528            new Class[] { interfaceClass },
529            new InvocationHandler() {
530
531               final RemoteableMeta rm = new RemoteableMeta(interfaceClass, restUrl2);
532
533               @Override /* InvocationHandler */
534               public Object invoke(Object proxy, Method method, Object[] args) throws Throwable {
535                  RemoteableMethodMeta rmm = rm.getMethodMeta(method);
536
537                  if (rmm == null)
538                     throw new RuntimeException("Method is not exposed as a remoteable method.");
539
540                  try {
541                     String url = rmm.getUrl();
542                     String httpMethod = rmm.getHttpMethod();
543                     RestCall rc = (httpMethod.equals("POST") ? doPost(url) : doGet(url));
544                     rc.serializer(serializer).parser(parser);
545
546                     for (RemoteMethodArg a : rmm.getPathArgs())
547                        rc.path(a.name, args[a.index], a.serializer);
548
549                     for (RemoteMethodArg a : rmm.getQueryArgs())
550                        rc.query(a.name, args[a.index], a.skipIfNE, a.serializer);
551
552                     for (RemoteMethodArg a : rmm.getFormDataArgs())
553                        rc.formData(a.name, args[a.index], a.skipIfNE, a.serializer);
554
555                     for (RemoteMethodArg a : rmm.getHeaderArgs())
556                        rc.header(a.name, args[a.index], a.skipIfNE, a.serializer);
557
558                     if (rmm.getBodyArg() != null)
559                        rc.input(args[rmm.getBodyArg()]);
560
561                     if (rmm.getRequestBeanArgs().length > 0) {
562                        BeanSession bs = getBeanContext().createSession();
563
564                        for (Integer i : rmm.getRequestBeanArgs()) {
565                           BeanMap<?> bm = bs.toBeanMap(args[i]);
566                           for (BeanPropertyValue bpv : bm.getValues(true)) {
567                              BeanPropertyMeta pMeta = bpv.getMeta();
568                              Object val = bpv.getValue();
569
570                              Path p = pMeta.getAnnotation(Path.class);
571                              if (p != null)
572                                 rc.path(getName(p.value(), pMeta), val, getPartSerializer(p.serializer()));
573
574                              Query q1 = pMeta.getAnnotation(Query.class);
575                              if (q1 != null)
576                                 rc.query(getName(q1.value(), pMeta), val, false, getPartSerializer(q1.serializer()));
577
578                              QueryIfNE q2 = pMeta.getAnnotation(QueryIfNE.class);
579                              if (q2 != null)
580                                 rc.query(getName(q2.value(), pMeta), val, true, getPartSerializer(q2.serializer()));
581
582                              FormData f1 = pMeta.getAnnotation(FormData.class);
583                              if (f1 != null)
584                                 rc.formData(getName(f1.value(), pMeta), val, false, getPartSerializer(f1.serializer()));
585
586                              FormDataIfNE f2 = pMeta.getAnnotation(FormDataIfNE.class);
587                              if (f2 != null)
588                                 rc.formData(getName(f2.value(), pMeta), val, true, getPartSerializer(f2.serializer()));
589
590                              org.apache.juneau.remoteable.Header h1 = pMeta.getAnnotation(org.apache.juneau.remoteable.Header.class);
591                              if (h1 != null)
592                                 rc.header(getName(h1.value(), pMeta), val, false, getPartSerializer(h1.serializer()));
593
594                              HeaderIfNE h2 = pMeta.getAnnotation(HeaderIfNE.class);
595                              if (h2 != null)
596                                 rc.header(getName(h2.value(), pMeta), val, true, getPartSerializer(h2.serializer()));
597                           }
598                        }
599                     }
600
601                     if (rmm.getOtherArgs().length > 0) {
602                        Object[] otherArgs = new Object[rmm.getOtherArgs().length];
603                        int i = 0;
604                        for (Integer otherArg : rmm.getOtherArgs())
605                           otherArgs[i++] = args[otherArg];
606                        rc.input(otherArgs);
607                     }
608
609                     return rc.getResponse(method.getGenericReturnType());
610
611                  } catch (RestCallException e) {
612                     // Try to throw original exception if possible.
613                     e.throwServerException(interfaceClass.getClassLoader());
614                     throw new RuntimeException(e);
615                  } catch (Exception e) {
616                     throw new RuntimeException(e);
617                  }
618               }
619         });
620      } catch (Exception e) {
621         throw new RuntimeException(e);
622      }
623   }

631   private static PartSerializer getPartSerializer(Class c) {
632      if (c == UrlEncodingSerializer.class)
633         return null;
634      PartSerializer pf = partSerializerCache.get(c);
635      if (pf == null) {
636         try {
637            partSerializerCache.putIfAbsent(c, (PartSerializer)c.newInstance());
638         } catch (Exception e) {
639            throw new RuntimeException(e);
640         }
641         pf = partSerializerCache.get(c);
642      }
643      return pf;
644   }

652   URI toURI(Object url) throws URISyntaxException {
653      if (url instanceof URI)
654         return (URI)url;
655      if (url instanceof URL)
656         ((URL)url).toURI();
657      if (url instanceof URIBuilder)
658         return ((URIBuilder)url).build();
659      String s = url == null ? "" : url.toString();
660      if (rootUrl != null && ! absUrlPattern.matcher(s).matches()) {
661         if (s.isEmpty())
662            s = rootUrl;
663         else {
664            StringBuilder sb = new StringBuilder(rootUrl);
665            if (! s.startsWith("/"))
666               sb.append('/');
667            sb.append(s);
668            s = sb.toString();
669         }
670      }
671      if (s.indexOf('{') != -1)
672         s = s.replace("{", "%7B").replace("}", "%7D");
673      return new URI(s);
674   }
