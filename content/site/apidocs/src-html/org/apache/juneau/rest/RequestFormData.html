023/**
024 * Represents the parsed form data parameters in an HTTP request.
025 */
026@SuppressWarnings("unchecked")
027public class RequestFormData extends LinkedHashMap<String,String[]> {
028   private static final long serialVersionUID = 1L;
029
030   private UrlEncodingParser parser;
031   private BeanSession beanSession;
032
033   RequestFormData setParser(UrlEncodingParser parser) {
034      this.parser = parser;
035      return this;
036   }
037
038   RequestFormData setBeanSession(BeanSession beanSession) {
039      this.beanSession = beanSession;
040      return this;
041   }

043   /**
044    * Adds default entries to these form-data parameters.
045    * <p>
046    * This includes the default form-data parameters defined on the servlet and method levels.
047    *
048    * @param defaultEntries The default entries.  Can be <jk>null</jk>.
049    * @return This object (for method chaining).
050    */
051   public RequestFormData addDefault(Map<String,String> defaultEntries) {
052      if (defaultEntries != null) {
053         for (Map.Entry<String,String> e : defaultEntries.entrySet()) {
054            String key = e.getKey(), value = e.getValue();
055            String[] v = get(key);
056            if (v == null || v.length == 0 || StringUtils.isEmpty(v[0]))
057               put(key, new String[]{value});
058         }
059      }
060      return this;
061   }
062
063   /**
064    * Sets a request form data parameter value.
065    *
066    * @param name The parameter name.
067    * @param value The parameter value.
068    */
069   public void put(String name, Object value) {
070      super.put(name, new String[]{StringUtils.toString(value)});
071   }
072
073   /**
074    * Returns a form data parameter value.
075    * <p>
076    * Parameter lookup is case-insensitive (consistent with WAS, but differs from Tomcat).
077    * <p>
078    * <h5 class='section'>Notes:</h5>
079    * <ul>
080    *    <li>Calling this method on URL-Encoded FORM posts causes the body content to be loaded and parsed by the underlying servlet API.
081    *    <li>This method returns the raw unparsed value, and differs from calling <code>getFormDataParameter(name, String.<jk>class</js>)</code>
082    *       which will convert the value from UON notation:
083    *       <ul>
084    *          <li><js>"null"</js> =&gt; <jk>null</jk>
085    *          <li><js>"'null'"</js> =&gt; <js>"null"</js>
086    *          <li><js>"'foo bar'"</js> =&gt; <js>"foo bar"</js>
087    *          <li><js>"foo~~bar"</js> =&gt; <js>"foo~bar"</js>
088    *       </ul>
089    * </ul>
090    *
091    * @param name The form data parameter name.
092    * @return The parameter value, or <jk>null</jk> if parameter does not exist.
093    */
094   public String getFirst(String name) {
095      String[] v = get(name);
096      if (v == null || v.length == 0)
097         return null;
098
099      // Fix for behavior difference between Tomcat and WAS.
100      // getParameter("foo") on "&foo" in Tomcat returns "".
101      // getParameter("foo") on "&foo" in WAS returns null.
102      if (v.length == 1 && v[0] == null) 
103         return "";
104
105      return v[0];
106   }
107
108   /**
109    * Same as {@link #getFirst(String)} except returns a default value if <jk>null</jk> or empty.
110    *
111    * @param name The form data parameter name.
112    * @param def The default value.
113    * @return The parameter value, or the default value if <jk>null</jk> or empty.
114    */
115   public String getFirst(String name, String def) {
116      String s = getFirst(name);
117      return StringUtils.isEmpty(s) ? def : s;
118   }

120   /**
121    * Returns the specified form data parameter value converted to a POJO using the
122    *    {@link UrlEncodingParser} registered with this servlet.
123    * <p>
124    * <h5 class='section'>Examples:</h5>
125    * <p class='bcode'>
126    *    <jc>// Parse into an integer.</jc>
127    *    <jk>int</jk> myparam = req.getFormDataParameter(<js>"myparam"</js>, <jk>int</jk>.<jk>class</jk>);
128    *
129    *    <jc>// Parse into an int array.</jc>
130    *    <jk>int</jk>[] myparam = req.getFormDataParameter(<js>"myparam"</js>, <jk>int</jk>[].<jk>class</jk>);
131
132    *    <jc>// Parse into a bean.</jc>
133    *    MyBean myparam = req.getFormDataParameter(<js>"myparam"</js>, MyBean.<jk>class</jk>);
134    *
135    *    <jc>// Parse into a linked-list of objects.</jc>
136    *    List myparam = req.getFormDataParameter(<js>"myparam"</js>, LinkedList.<jk>class</jk>);
137    *
138    *    <jc>// Parse into a map of object keys/values.</jc>
139    *    Map myparam = req.getFormDataParameter(<js>"myparam"</js>, TreeMap.<jk>class</jk>);
140    * </p>
141    * <p>
142    * <h5 class='section'>Notes:</h5>
143    * <ul>
144    *    <li>Calling this method on URL-Encoded FORM posts causes the body content to be loaded and parsed by the underlying servlet API.
145    * </ul>
146    *
147    * @param name The parameter name.
148    * @param type The class type to convert the parameter value to.
149    * @param <T> The class type to convert the parameter value to.
150    * @return The parameter value converted to the specified class type.
151    * @throws ParseException
152    */
153   public <T> T get(String name, Class<T> type) throws ParseException {
154      return parse(name, beanSession.getClassMeta(type));
155   }
156
157   /**
158    * Same as {@link #get(String, Class)} except returns a default value if not specified.
159    *
160    * @param name The parameter name.
161    * @param def The default value if the parameter was not specified or is <jk>null</jk>.
162    * @param type The class type to convert the parameter value to.
163    * @param <T> The class type to convert the parameter value to.
164    * @return The parameter value converted to the specified class type.
165    * @throws ParseException
166    */
167   public <T> T get(String name, T def, Class<T> type) throws ParseException {
168      return parse(name, def, beanSession.getClassMeta(type));
169   }

171   /**
172    * Same as {@link #get(String, Class)} except for use on multi-part parameters
173    *    (e.g. <js>"key=1&amp;key=2&amp;key=3"</js> instead of <js>"key=(1,2,3)"</js>)
174    * <p>
175    * This method must only be called when parsing into classes of type Collection or array.
176    *
177    * @param name The parameter name.
178    * @param type The class type to convert the parameter value to.
179    * @return The parameter value converted to the specified class type.
180    * @throws ParseException
181    */
182   public <T> T getAll(String name, Class<T> type) throws ParseException {
183      return parseAll(name, beanSession.getClassMeta(type));
184   }

186   /**
187    * Returns the specified form data parameter value converted to a POJO using the
188    *    {@link UrlEncodingParser} registered with this servlet.
189    * <p>
190    * <h5 class='section'>Notes:</h5>
191    * <ul>
192    *    <li>Calling this method on URL-Encoded FORM posts causes the body content to be loaded and parsed by the underlying servlet API.
193    *    <li>Use this method if you want to parse into a parameterized <code>Map</code>/<code>Collection</code> object.
194    * </ul>
195    * <p>
196    * <h5 class='section'>Examples:</h5>
197    * <p class='bcode'>
198    *    <jc>// Parse into a linked-list of strings.</jc>
199    *    List&lt;String&gt; myparam = req.getFormDataParameter(<js>"myparam"</js>, LinkedList.<jk>class</jk>, String.<jk>class</jk>);
200    *
201    *    <jc>// Parse into a linked-list of linked-lists of strings.</jc>
202    *    List&lt;List&lt;String&gt;&gt; myparam = req.getFormDataParameter(<js>"myparam"</js>, LinkedList.<jk>class</jk>, LinkedList.<jk>class</jk>, String.<jk>class</jk>);
203    *
204    *    <jc>// Parse into a map of string keys/values.</jc>
205    *    Map&lt;String,String&gt; myparam = req.getFormDataParameter(<js>"myparam"</js>, TreeMap.<jk>class</jk>, String.<jk>class</jk>, String.<jk>class</jk>);
206    *
207    *    <jc>// Parse into a map containing string keys and values of lists containing beans.</jc>
208    *    Map&lt;String,List&lt;MyBean&gt;&gt; myparam = req.getFormDataParameter(<js>"myparam"</js>, TreeMap.<jk>class</jk>, String.<jk>class</jk>, List.<jk>class</jk>, MyBean.<jk>class</jk>);
209    * </p>
210    *
211    * @param name The parameter name.
212    * @param type The type of object to create.
213    *    <br>Can be any of the following: {@link ClassMeta}, {@link Class}, {@link ParameterizedType}, {@link GenericArrayType}
214    * @param args The type arguments of the class if it's a collection or map.
215    *    <br>Can be any of the following: {@link ClassMeta}, {@link Class}, {@link ParameterizedType}, {@link GenericArrayType}
216    *    <br>Ignored if the main type is not a map or collection.
217    * @return The parameter value converted to the specified class type.
218    * @throws ParseException
219    */
220   public <T> T get(String name, Type type, Type...args) throws ParseException {
221      return (T)parse(name, beanSession.getClassMeta(type, args));
222   }
223
224   /**
225    * Same as {@link #get(String, Type, Type...)} except for use on multi-part parameters
226    *    (e.g. <js>"key=1&amp;key=2&amp;key=3"</js> instead of <js>"key=(1,2,3)"</js>)
227    * <p>
228    * This method must only be called when parsing into classes of type Collection or array.
229    *
230    * @param name The parameter name.
231    * @param type The type of object to create.
232    *    <br>Can be any of the following: {@link ClassMeta}, {@link Class}, {@link ParameterizedType}, {@link GenericArrayType}
233    * @param args The type arguments of the class if it's a collection or map.
234    *    <br>Can be any of the following: {@link ClassMeta}, {@link Class}, {@link ParameterizedType}, {@link GenericArrayType}
235    *    <br>Ignored if the main type is not a map or collection.
236    * @return The parameter value converted to the specified class type.
237    * @throws ParseException
238    */
239   public <T> T getAll(String name, Type type, Type...args) throws ParseException {
240      return (T)parseAll(name, beanSession.getClassMeta(type, args));
241   }
242
243   /* Workhorse method */
244   <T> T parse(String name, T def, ClassMeta<T> cm) throws ParseException {
245      String val = getFirst(name);
246      if (val == null)
247         return def;
248      return parseValue(val, cm);
249   }
250
251   /* Workhorse method */
252   <T> T parse(String name, ClassMeta<T> cm) throws ParseException {
253      String val = getFirst(name);
254      if (cm.isPrimitive() && (val == null || val.isEmpty()))
255         return cm.getPrimitiveDefault();
256      return parseValue(val, cm);
257   }
258
259   /* Workhorse method */
260   @SuppressWarnings("rawtypes")
261   <T> T parseAll(String name, ClassMeta<T> cm) throws ParseException {
262      String[] p = get(name);
263      if (p == null)
264         return null;
265      if (cm.isArray()) {
266         List c = new ArrayList();
267         for (int i = 0; i < p.length; i++)
268            c.add(parseValue(p[i], cm.getElementType()));
269         return (T)ArrayUtils.toArray(c, cm.getElementType().getInnerClass());
270      } else if (cm.isCollection()) {
271         try {
272            Collection c = (Collection)(cm.canCreateNewInstance() ? cm.newInstance() : new ObjectList());
273            for (int i = 0; i < p.length; i++)
274               c.add(parseValue(p[i], cm.getElementType()));
275            return (T)c;
276         } catch (ParseException e) {
277            throw e;
278         } catch (Exception e) {
279            // Typically an instantiation exception.
280            throw new ParseException(e);
281         }
282      }
283      throw new ParseException("Invalid call to getParameters(String, ClassMeta).  Class type must be a Collection or array.");
284   }
285
286   private <T> T parseValue(String val, ClassMeta<T> c) throws ParseException {
287      return parser.parsePart(val, c);
288   }
