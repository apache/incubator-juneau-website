052    * <b>Configuration property:</b>  Enable header URL parameters.
053    * <p>
054    * <ul>
055    *    <li><b>Name:</b> <js>"RestServlet.allowHeaderParams"</js>
056    *    <li><b>Data type:</b> <code>Boolean</code>

058    * </ul>
059    * <p>
060    * When enabled, headers such as <js>"Accept"</js> and <js>"Content-Type"</js> to be passed in as URL query parameters.
061    * For example:  <js>"?Accept=text/json&amp;Content-Type=text/json"</js>
062    * <p>
063    * Parameter names are case-insensitive.
064    * <p>
065    * Useful for debugging REST interface using only a browser.
066    * <p>
067    * Applicable to servlet class only.
068    */

071   /**
072    * <b>Configuration property:</b>  Enable <js>"method"</js> URL parameter for specific HTTP methods.
073    * <p>
074    * <ul>
075    *    <li><b>Name:</b> <js>"RestServlet.allowMethodParam"</js>
076    *    <li><b>Data type:</b> <code>String</code>
077    *    <li><b>Default:</b> <js>""</js>
078    * </ul>
079    * <p>
080    * When specified, the HTTP method can be overridden by passing in a <js>"method"</js> URL parameter on a regular GET request.
081    * For example:  <js>"?method=OPTIONS"</js>
082    * <p>
083    * Format is a comma-delimited list of HTTP method names that can be passed in as a method parameter.
084    * Parameter name is case-insensitive.
085    * Use "*" to represent all methods.
086    * For backwards compatibility, "true" also means "*".
087    * <p>
088    * Note that per the <a class="doclink" href="http://www.w3.org/Protocols/rfc2616/rfc2616-sec9.html">HTTP specification</a>, special care should
089    *    be taken when allowing non-safe (POST, PUT, DELETE) methods to be invoked through GET requests.
090    * <p>
091    * Applicable to servlet class only.
092    * <p>
093    * Example: <js>"HEAD,OPTIONS"</js>
094    */
095   public static final String REST_allowMethodParam = "RestServlet.allowMethodParam";

343   /**
344    * Constructor.
345    *
346    * @param resource The resource class (a class annotated with {@link RestResource @RestResource}).
347    * @param config The servlet configuration object.
348    * @throws Exception If any initialization problems were encountered.
349    */
350   @SuppressWarnings("unchecked")
351   public RestContext(Object resource, RestConfig config) throws Exception {
352      super(null);
353      RestException _initException = null;
354      try {
355         this.resource = resource;
356         this.config = config;
357         this.parentContext = config.parentContext;
358
359         Builder b = new Builder(resource, config);
360         this.allowHeaderParams = b.allowHeaderParams;
361         this.allowBodyParam = b.allowBodyParam;
362         this.renderResponseStackTraces = b.renderResponseStackTraces;
363         this.useStackTraceHashes = b.useStackTraceHashes;
364         this.allowMethodParams = Collections.unmodifiableSet(b.allowMethodParams);
365         this.defaultCharset = b.defaultCharset;
366         this.paramFormat = b.paramFormat;
367         this.varResolver = b.varResolver;
368         this.configFile = b.configFile;
369         this.properties = b.properties;
370         this.beanFilters = b.beanFilters;
371         this.pojoSwaps = b.pojoSwaps;
372         this.paramResolvers = Collections.unmodifiableMap(b.paramResolvers);
373         this.serializers = b.serializers;
374         this.parsers = b.parsers;
375         this.urlEncodingSerializer = b.urlEncodingSerializer;
376         this.urlEncodingParser = b.urlEncodingParser;
377         this.encoders = b.encoders;
378         this.supportedContentTypes = ArrayUtils.toObjectArray(b.supportedContentTypes, MediaType.class);
379         this.supportedAcceptTypes = ArrayUtils.toObjectArray(b.supportedAcceptTypes, MediaType.class);
380         this.clientVersionHeader = b.clientVersionHeader;
381         this.defaultRequestHeaders = Collections.unmodifiableMap(b.defaultRequestHeaders);
382         this.defaultResponseHeaders = Collections.unmodifiableMap(b.defaultResponseHeaders);
383         this.beanContext = b.beanContext;
384         this.converters = b.converters.toArray(new RestConverter[b.converters.size()]);
385         this.guards = b.guards.toArray(new RestGuard[b.guards.size()]);
386         this.responseHandlers = ArrayUtils.toObjectArray(b.responseHandlers, ResponseHandler.class);
387         this.mimetypesFileTypeMap = b.mimetypesFileTypeMap;
388         this.styleSheet = b.styleSheet;
389         this.favIcon = b.favIcon;
390         this.staticFilesMap = Collections.unmodifiableMap(b.staticFilesMap);
391         this.staticFilesPrefixes = b.staticFilesPrefixes;
392         this.msgs = b.messageBundle;
393         this.childResources = Collections.synchronizedMap(new LinkedHashMap<String,RestContext>());  // Not unmodifiable on purpose so that children can be replaced.
394         this.logger = b.logger;
395         this.fullPath = b.fullPath;
396         this.pageTitle = b.pageTitle;
397         this.pageText = b.pageText;
398         this.pageLinks = b.pageLinks;
399
400         //----------------------------------------------------------------------------------------------------
401         // Initialize the child resources.
402         // Done after initializing fields above since we pass this object to the child resources.
403         //----------------------------------------------------------------------------------------------------
404         List<String> methodsFound = new LinkedList<String>();   // Temporary to help debug transient duplicate method issue.
405         Map<String,CallRouter.Builder> routers = new LinkedHashMap<String,CallRouter.Builder>();
406         Map<String,CallMethod> _javaRestMethods = new LinkedHashMap<String,CallMethod>();
407         for (java.lang.reflect.Method method : resource.getClass().getMethods()) {
408            if (method.isAnnotationPresent(RestMethod.class)) {
409               RestMethod a = method.getAnnotation(RestMethod.class);
410               methodsFound.add(method.getName() + "," + a.name() + "," + a.path());
411               try {
412                  if (! Modifier.isPublic(method.getModifiers()))
413                     throw new RestServletException("@RestMethod method {0}.{1} must be defined as public.", this.getClass().getName(), method.getName());
414
415                  CallMethod sm = new CallMethod(resource, method, this);
416                  String httpMethod = sm.getHttpMethod();
417
418                  // PROXY is a special case where a method returns an interface that we
419                  // can perform REST calls against.
420                  // We override the CallMethod.invoke() method to insert our logic.
421                  if ("PROXY".equals(httpMethod)) {
422
423                     final ClassMeta<?> interfaceClass = beanContext.getClassMeta(method.getGenericReturnType());
424                     final Map<String,Method> remoteableMethods = interfaceClass.getRemoteableMethods();
425                     if (remoteableMethods.isEmpty())
426                        throw new RestException(SC_INTERNAL_SERVER_ERROR, "Method {0} returns an interface {1} that doesn't define any remoteable methods.", ClassUtils.getMethodSignature(method), interfaceClass.getReadableName());
427
428                     sm = new CallMethod(resource, method, this) {
429
430                        @Override
431                        int invoke(String pathInfo, RestRequest req, RestResponse res) throws RestException {
432
433                           int rc = super.invoke(pathInfo, req, res);
434                           if (rc != SC_OK)
435                              return rc;
436
437                           final Object o = res.getOutput();
438
439                           if ("GET".equals(req.getMethod())) {
440                              res.setOutput(ClassUtils.getMethodInfo(remoteableMethods.values()));
441                              return SC_OK;
442
443                           } else if ("POST".equals(req.getMethod())) {
444                              if (pathInfo.indexOf('/') != -1)
445                                 pathInfo = pathInfo.substring(pathInfo.lastIndexOf('/')+1);
446                              pathInfo = urlDecode(pathInfo);
447                              java.lang.reflect.Method m = remoteableMethods.get(pathInfo);
448                              if (m != null) {
449                                 try {
450                                    // Parse the args and invoke the method.
451                                    Parser p = req.getBody().getParser();
452                                    Object input = p.isReaderParser() ? req.getReader() : req.getInputStream();
453                                    Object output = m.invoke(o, p.parseArgs(input, m.getGenericParameterTypes()));
454                                    res.setOutput(output);
455                                    return SC_OK;
456                                 } catch (Exception e) {
457                                    throw new RestException(SC_INTERNAL_SERVER_ERROR, e);
458                                 }
459                              }
460                           }
461                           return SC_NOT_FOUND;
462                        }
463                     };
464
465                     _javaRestMethods.put(method.getName(), sm);
466                     addToRouter(routers, "GET", sm);
467                     addToRouter(routers, "POST", sm);
468
469                  } else {
470                     _javaRestMethods.put(method.getName(), sm);
471                     addToRouter(routers, httpMethod, sm);
472                  }
473               } catch (RestServletException e) {
474                  throw new RestServletException("Problem occurred trying to serialize methods on class {0}, methods={1}", this.getClass().getName(), JsonSerializer.DEFAULT_LAX.serialize(methodsFound)).initCause(e);
475               }
476            }
477         }
478         this.callMethods = Collections.unmodifiableMap(_javaRestMethods);
479
480         Map<String,CallRouter> _callRouters = new LinkedHashMap<String,CallRouter>();
481         for (CallRouter.Builder crb : routers.values())
482            _callRouters.put(crb.getHttpMethodName(), crb.build());
483         this.callRouters = Collections.unmodifiableMap(_callRouters);
484
485         // Initialize our child resources.
486         RestResourceResolver rrr = resolve(RestResourceResolver.class, config.resourceResolver);
487         for (Object o : config.childResources) {
488            String path = null;
489            Object r = null;
490            if (o instanceof Pair) {
491               Pair<String,Object> p = (Pair<String,Object>)o;
492               path = p.first();
493               r = p.second();
494            } else if (o instanceof Class<?>) {
495               Class<?> c = (Class<?>)o;
496               r = c;
497            } else {
498               r = o;
499            }
500
501            RestConfig childConfig = null;
502
503            if (o instanceof Class) {
504               Class<?> oc = (Class<?>)o;
505               childConfig = new RestConfig(config.inner, oc, this);
506               r = rrr.resolve(oc, childConfig);
507            } else {
508               r = o;
509               childConfig = new RestConfig(config.inner, o.getClass(), this);
510            }
511
512            if (r instanceof RestServlet) {
513               RestServlet rs = (RestServlet)r;
514               rs.init(childConfig);
515               path = childConfig.path;
516               childResources.put(path, rs.getContext());
517            } else {
518
519               // Call the init(RestConfig) method.
520               java.lang.reflect.Method m2 = ClassUtils.findPublicMethod(r.getClass(), "init", Void.class, RestConfig.class);
521               if (m2 != null)
522                  m2.invoke(r, childConfig);
523
524               RestContext rc2 = new RestContext(r, childConfig);
525
526               // Call the init(RestContext) method.
527               m2 = ClassUtils.findPublicMethod(r.getClass(), "init", Void.class, RestContext.class);
528               if (m2 != null)
529                  m2.invoke(r, rc2);
530
531               path = childConfig.path;
532               childResources.put(path, rc2);
533            }
534         }
535
536         callHandler = config.callHandler == null ? new RestCallHandler(this) : resolve(RestCallHandler.class, config.callHandler, this);
537         infoProvider = config.infoProvider == null ? new RestInfoProvider(this) : resolve(RestInfoProvider.class, config.infoProvider, this);
538
539      } catch (RestException e) {
540         _initException = e;
541         throw e;
542      } catch (Exception e) {
543         _initException = new RestException(SC_INTERNAL_SERVER_ERROR, e);
544         throw e;
545      } finally {
546         initException = _initException;
547      }
548   }

550   private static void addToRouter(Map<String, CallRouter.Builder> routers, String httpMethodName, CallMethod cm) throws RestServletException {
551      if (! routers.containsKey(httpMethodName))
552         routers.put(httpMethodName, new CallRouter.Builder(httpMethodName));
553      routers.get(httpMethodName).add(cm);
554   }

780    * Resolve a static resource file.
781    * <p>
782    * The location of static resources are defined via one of the following:
783    * <ul>
784    *    <li>{@link RestResource#staticFiles() @RestResource.staticFiles()} annotation.
785    *    <li>{@link RestConfig#addStaticFiles(Class, String)} method.
786    * </ul>
787    *
788    * @param pathInfo The unencoded path info.
789    * @return The resource, or <jk>null</jk> if the resource could not be resolved.
790    * @throws IOException
791    */
792   public StreamResource resolveStaticFile(String pathInfo) throws IOException {
793      if (! staticFilesCache.containsKey(pathInfo)) {
794         String p = urlDecode(trimSlashes(pathInfo));
795         if (p.indexOf("..") != -1)
796            throw new RestException(SC_NOT_FOUND, "Invalid path");
797         for (Map.Entry<String,String> e : staticFilesMap.entrySet()) {
798            String key = trimSlashes(e.getKey());
799            if (p.startsWith(key)) {
800               String remainder = (p.equals(key) ? "" : p.substring(key.length()));
801               if (remainder.isEmpty() || remainder.startsWith("/")) {
802                  String p2 = trimSlashes(e.getValue()) + remainder;
803                  InputStream is = getResource(p2, null);
804                  if (is != null) {
805                     try {
806                        int i = p2.lastIndexOf('/');
807                        String name = (i == -1 ? p2 : p2.substring(i+1));
808                        String mediaType = mimetypesFileTypeMap.getContentType(name);
809                        ObjectMap headers = new ObjectMap().append("Cache-Control", "max-age=86400, public");
810                        staticFilesCache.put(pathInfo, new StreamResource(MediaType.forString(mediaType), headers, is));
811                        return staticFilesCache.get(pathInfo);
812                     } finally {
813                        is.close();
814                     }
815                  }
816               }
817            }
818         }
819      }
820      return staticFilesCache.get(pathInfo);
821   }

824    * Same as {@link Class#getResourceAsStream(String)} except if it doesn't find the resource
825    *    on this class, searches up the parent hierarchy chain.
826    * <p>
827    * If the resource cannot be found in the classpath, then an attempt is made to look in the
828    *    JVM working directory.
829    * <p>
830    * If the <code>locale</code> is specified, then we look for resources whose name matches that locale.
831    * For example, if looking for the resource <js>"MyResource.txt"</js> for the Japanese locale, we will
832    *    look for files in the following order:
833    * <ol>
834    *    <li><js>"MyResource_ja_JP.txt"</js>
835    *    <li><js>"MyResource_ja.txt"</js>
836    *    <li><js>"MyResource.txt"</js>
837    * </ol>
838    *
839    * @param name The resource name.
840    * @param locale Optional locale.
841    * @return An input stream of the resource, or <jk>null</jk> if the resource could not be found.
842    * @throws IOException
843    */
844   protected InputStream getResource(String name, Locale locale) throws IOException {
845      String n = (locale == null || locale.toString().isEmpty() ? name : name + '|' + locale);
846      if (! resourceStreams.containsKey(n)) {
847         InputStream is = ReflectionUtils.getLocalizedResource(resource.getClass(), name, locale);
848         if (is == null && name.indexOf("..") == -1) {
849            for (String n2 : FileUtils.getCandidateFileNames(name, locale)) {
850               File f = new File(n2);
851               if (f.exists() && f.canRead()) {
852                  is = new FileInputStream(f);
853                  break;
854               }
855            }
856         }
857         if (is != null) {
858            try {
859               resourceStreams.put(n, ByteArrayCache.DEFAULT.cache(is));
860            } finally {
861               is.close();
862            }
863         }
864      }
865      byte[] b = resourceStreams.get(n);
866      return b == null ? null : new ByteArrayInputStream(b);
867   }

892    * Useful if you want to load predefined POJOs from JSON files in your classpath.

903      InputStream is = getResource(name, locale);

913            } catch (ParseException e) {
914               throw new ServletException("Could not parse resource '' as media type '"+mediaType+"'.");
915            }
916         }
917         throw new ServletException("Unknown media type '"+mediaType+"'");
918      } catch (Exception e) {

1267   /**
1268    * Finds the {@link RestParam} instances to handle resolving objects on the calls to the specified Java method.
1269    *
1270    * @param method The Java method being called.
1271    * @param methodPlainParams Whether plain-params setting is specified.
1272    * @param pathPattern The parsed URL path pattern.
1273    * @return The array of resolvers.
1274    * @throws ServletException If an annotation usage error was detected.
1275    */
1276   protected RestParam[] findParams(Method method, boolean methodPlainParams, UrlPathPattern pathPattern) throws ServletException {
1277
1278      Type[] pt = method.getGenericParameterTypes();
1279      Annotation[][] pa = method.getParameterAnnotations();
1280      RestParam[] rp = new RestParam[pt.length];
1281      int attrIndex = 0;
1282
1283      for (int i = 0; i < pt.length; i++) {
1284
1285         Type t = pt[i];
1286         if (t instanceof Class) {
1287            Class<?> c = (Class<?>)t;
1288            rp[i] = paramResolvers.get(c);
1289            if (rp[i] == null)
1290               rp[i] = RestParamDefaults.STANDARD_RESOLVERS.get(c);
1291         }
1292
1293         if (rp[i] == null) {
1294            for (Annotation a : pa[i]) {
1295               if (a instanceof Header)
1296                  rp[i] = new RestParamDefaults.HeaderObject((Header)a, t);
1297               else if (a instanceof FormData)
1298                  rp[i] = new RestParamDefaults.FormDataObject(method, (FormData)a, t, methodPlainParams);
1299               else if (a instanceof Query)
1300                  rp[i] = new RestParamDefaults.QueryObject(method, (Query)a, t, methodPlainParams);
1301               else if (a instanceof HasFormData)
1302                  rp[i] = new RestParamDefaults.HasFormDataObject(method, (HasFormData)a, t);
1303               else if (a instanceof HasQuery)
1304                  rp[i] = new RestParamDefaults.HasQueryObject(method, (HasQuery)a, t);
1305               else if (a instanceof Body)
1306                  rp[i] = new RestParamDefaults.BodyObject(t);
1307               else if (a instanceof org.apache.juneau.rest.annotation.Method)
1308                  rp[i] = new RestParamDefaults.MethodObject(method, t);
1309               else if (a instanceof PathRemainder)
1310                  rp[i] = new RestParamDefaults.PathRemainderObject(method, t);
1311               else if (a instanceof Properties)
1312                  rp[i] = new RestParamDefaults.PropsObject(method, t);
1313               else if (a instanceof Messages)
1314                  rp[i] = new RestParamDefaults.MessageBundleObject();
1315            }
1316         }
1317
1318         if (rp[i] == null) {
1319            Path p = null;
1320            for (Annotation a : pa[i])
1321               if (a instanceof Path)
1322                  p = (Path)a;
1323
1324            String name = (p == null ? "" : p.value());
1325
1326            if (name.isEmpty()) {
1327               int idx = attrIndex++;
1328               String[] vars = pathPattern.getVars();
1329               if (vars.length <= idx)
1330                  throw new RestServletException("Number of attribute parameters in method ''{0}'' exceeds the number of URL pattern variables.", method.getName());
1331
1332               // Check for {#} variables.
1333               String idxs = String.valueOf(idx);
1334               for (int j = 0; j < vars.length; j++)
1335                  if (StringUtils.isNumeric(vars[j]) && vars[j].equals(idxs))
1336                     name = vars[j];
1337
1338               if (name.isEmpty())
1339                  name = pathPattern.getVars()[idx];
1340            }
1341            rp[i] = new RestParamDefaults.PathParameterObject(name, t);
1342         }
1343      }
1344
1345      return rp;
1346   }

1600   /**
1601    * Takes in an object of type T or a Class<T> and either casts or constructs a T.
1602    */
1603   @SuppressWarnings("unchecked")
1604   private static <T> T resolve(Class<T> c, Object o, Object...cArgs) throws RestServletException {
1605      if (c.isInstance(o))
1606         return (T)o;
1607      if (! (o instanceof Class))
1608         throw new RestServletException("Invalid object type passed to resolve:  ''{0}''.  Must be an object of type T or a Class<? extend T>.", o.getClass());
1609      Constructor<T> n = ClassUtils.findPublicConstructor((Class<T>)o, cArgs);
1610      if (n == null)
1611         throw new RestServletException("Could not find public constructor for class ''{0}'' that takes in args {1}", c, JsonSerializer.DEFAULT_LAX.toString(ClassUtils.getClasses(cArgs)));
1612      try {
1613         return n.newInstance(cArgs);
1614      } catch (Exception e) {
1615         throw new RestServletException("Exception occurred while constructing class ''{0}''", c).initCause(e);
1616      }
1617   }
